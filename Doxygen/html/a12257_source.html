41 Foam::polynomial::polynomial(const word& entryName, const dictionary& dict)
42 :
43  scalarDataEntry(entryName),
44  coeffs_(),
45  canIntegrate_(true),
46  dimensions_(dimless)
47 {
48  Istream& is(dict.lookup(entryName));
49  word entryType(is);
50 
51  token firstToken(is);
52  is.putBack(firstToken);
53  if (firstToken == token::BEGIN_SQR)
54  {
55  is >> this->dimensions_;
56  }
57 
58  is >> coeffs_;
59 
60  if (!coeffs_.size())
61  {
62  FatalErrorIn
63  (
64  "Foam::polynomial::polynomial(const word&, const dictionary&)"
65  ) << "polynomial coefficients for entry " << this->name_
66  << " are invalid (empty)" << nl << exit(FatalError);
67  }
68 
69  forAll(coeffs_, i)
70  {
71  if (mag(coeffs_[i].second() + 1) < ROOTVSMALL)
72  {
73  canIntegrate_ = false;
74  break;
75  }
76  }
77 
78  if (debug)
79  {
80  if (!canIntegrate_)
81  {
82  WarningIn
83  (
84  "Foam::polynomial::polynomial(const word&, const dictionary&)"
85  ) << "Polynomial " << this->name_ << " cannot be integrated"
86  << endl;
87  }
88  }

176 Foam::scalar Foam::polynomial::integrate(const scalar x1, const scalar x2) const
177 {
178  scalar intx = 0.0;
179 
180  if (canIntegrate_)
181  {
182  forAll(coeffs_, i)
183  {
184  intx +=
185  coeffs_[i].first()/(coeffs_[i].second() + 1)
186  *(
187  pow(x2, coeffs_[i].second() + 1)
188  - pow(x1, coeffs_[i].second() + 1)
189  );
190  }
191  }
192 
193  return intx;
194 }
