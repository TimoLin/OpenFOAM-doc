61 tmp<scalarField> nutUSpaldingWallFunctionFvPatchScalarField::calcUTau
62 (
63  const scalarField& magGradU
64 ) const
65 {
66  const label patchI = patch().index();
67 
68  const turbulenceModel& turbModel =
69  db().lookupObject<turbulenceModel>("turbulenceModel");
70  const scalarField& y = turbModel.y()[patchI];
71 
72  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchI];
73  const scalarField magUp(mag(Uw.patchInternalField() - Uw));
74 
75  const tmp<volScalarField> tnu = turbModel.nu();
76  const volScalarField& nu = tnu();
77  const scalarField& nuw = nu.boundaryField()[patchI];
78 
79  const scalarField& nutw = *this;
80 
81  tmp<scalarField> tuTau(new scalarField(patch().size(), 0.0));
82  scalarField& uTau = tuTau();
83 
84  forAll(uTau, faceI)
85  {
86  scalar ut = sqrt((nutw[faceI] + nuw[faceI])*magGradU[faceI]);
87 
88  if (ut > ROOTVSMALL)
89  {
90  int iter = 0;
91  scalar err = GREAT;
92 
93  do
94  {
95  scalar kUu = min(kappa_*magUp[faceI]/ut, 50);
96  scalar fkUu = exp(kUu) - 1 - kUu*(1 + 0.5*kUu);
97 
98  scalar f =
99  - ut*y[faceI]/nuw[faceI]
100  + magUp[faceI]/ut
101  + 1/E_*(fkUu - 1.0/6.0*kUu*sqr(kUu));
102 
103  scalar df =
104  y[faceI]/nuw[faceI]
105  + magUp[faceI]/sqr(ut)
106  + 1/E_*kUu*fkUu/ut;
107 
108  scalar uTauNew = ut + f/df;
109  err = mag((ut - uTauNew)/ut);
110  ut = uTauNew;
111 
112  } while (ut > ROOTVSMALL && err > 0.01 && ++iter < 10);
113 
114  uTau[faceI] = max(0.0, ut);
115  }
116  }
117 
118  return tuTau;
119 }
