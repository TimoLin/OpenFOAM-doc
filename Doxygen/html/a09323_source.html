54 Foam::List<bool> Foam::timeSelector::selected(const instantList& Times) const
55 {
56  List<bool> lst(Times.size(), false);
57 
58  // Check ranges, avoid false positive on constant/
59  forAll(Times, timeI)
60  {
61  if (Times[timeI].name() != "constant" && selected(Times[timeI]))
62  {
63  lst[timeI] = true;
64  }
65  }
66 
67  // Check specific values
68  forAll(*this, rangeI)
69  {
70  if (operator[](rangeI).isExact())
71  {
72  scalar target = operator[](rangeI).value();
73 
74  int nearestIndex = -1;
75  scalar nearestDiff = Foam::GREAT;
76 
77  forAll(Times, timeI)
78  {
79  if (Times[timeI].name() == "constant") continue;
80 
81  scalar diff = fabs(Times[timeI].value() - target);
82  if (diff < nearestDiff)
83  {
84  nearestDiff = diff;
85  nearestIndex = timeI;
86  }
87  }
88 
89  if (nearestIndex >= 0)
90  {
91  lst[nearestIndex] = true;
92  }
93  }
94  }
95 
96  return lst;
97 }

155 Foam::List<Foam::instant> Foam::timeSelector::select
156 (
157  const instantList& timeDirs,
158  const argList& args,
159  const word& constantName
160 )
161 {
162  if (timeDirs.size())
163  {
164  List<bool> selectTimes(timeDirs.size(), true);
165 
166  // Determine locations of constant/ and 0/ directories
167  label constantIdx = -1;
168  label zeroIdx = -1;
169 
170  forAll(timeDirs, timeI)
171  {
172  if (timeDirs[timeI].name() == constantName)
173  {
174  constantIdx = timeI;
175  }
176  else if (timeDirs[timeI].value() == 0)
177  {
178  zeroIdx = timeI;
179  }
180 
181  if (constantIdx >= 0 && zeroIdx >= 0)
182  {
183  break;
184  }
185  }
186 
187  // Determine latestTime selection (if any)
188  // This must appear before the -time option processing
189  label latestIdx = -1;
190  if (args.optionFound("latestTime"))
191  {
192  selectTimes = false;
193  latestIdx = timeDirs.size() - 1;
194 
195  // Avoid false match on constant/
196  if (latestIdx == constantIdx)
197  {
198  latestIdx = -1;
199  }
200  }
201 
202  if (args.optionFound("time"))
203  {
204  // Can match 0/, but can never match constant/
205  selectTimes = timeSelector
206  (
207  args.optionLookup("time")()
208  ).selected(timeDirs);
209  }
210 
211  // Add in latestTime (if selected)
212  if (latestIdx >= 0)
213  {
214  selectTimes[latestIdx] = true;
215  }
216 
217  if (constantIdx >= 0)
218  {
219  // Only add constant/ if specifically requested
220  selectTimes[constantIdx] = args.optionFound("constant");
221  }
222 
223  // Special treatment for 0/
224  if (zeroIdx >= 0)
225  {
226  if (args.optionFound("noZero"))
227  {
228  // Exclude 0/ if specifically requested
229  selectTimes[zeroIdx] = false;
230  }
231  else if (argList::validOptions.found("withZero"))
232  {
233  // With -withZero enabled, drop 0/ unless specifically requested
234  selectTimes[zeroIdx] = args.optionFound("withZero");
235  }
236  }
237 
238  return subset(selectTimes, timeDirs);
239  }
240  else
241  {
242  return timeDirs;
243  }
244 }

247 Foam::List<Foam::instant> Foam::timeSelector::select0
248 (
249  Time& runTime,
250  const argList& args
251 )
252 {
253  instantList timeDirs = timeSelector::select
254  (
255  runTime.times(),
256  args,
257  runTime.constant()
258  );
259 
260  if (timeDirs.empty())
261  {
262  WarningIn(args.executable())
263  << "No time specified or available, selecting 'constant'"
264  << endl;
265 
266  timeDirs.append(instant(0, runTime.constant()));
267  }
268 
269  runTime.setTime(timeDirs[0], 0);
270 
271  return timeDirs;
272 }

275 Foam::List<Foam::instant> Foam::timeSelector::selectIfPresent
276 (
277  Time& runTime,
278  const argList& args
279 )
280 {
281  if
282  (
283  args.optionFound("latestTime")
284  || args.optionFound("time")
285  || args.optionFound("constant")
286  || args.optionFound("noZero")
287  || args.optionFound("withZero")
288  )
289  {
290  return select0(runTime, args);
291  }
292  else
293  {
294  // No timeSelector option specified. Do not change runTime.
295  return instantList(1, instant(runTime.value(), runTime.timeName()));
296  }
297 }
