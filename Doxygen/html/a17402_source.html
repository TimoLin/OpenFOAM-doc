116 void dynLagrangian::correct(const tmp<volTensorField>& gradU)
117 {
118  LESModel::correct(gradU);
119 
120  volSymmTensorField S(dev(symm(gradU())));
121 
122  volScalarField magS(mag(S));
123 
124  volVectorField Uf(filter_(U()));
125 
126  volSymmTensorField Sf(dev(symm(fvc::grad(Uf))));
127 
128  volScalarField magSf(mag(Sf));
129 
130  volSymmTensorField L(dev(filter_(sqr(U())) - (sqr(filter_(U())))));
131 
132  volSymmTensorField M(2.0*sqr(delta())*(filter_(magS*S) - 4.0*magSf*Sf));
133 
134  volScalarField invT
135  (
136  (1.0/(theta_.value()*delta()))*pow(flm_*fmm_, 1.0/8.0)
137  );
138 
139  volScalarField LM(L && M);
140 
141  fvScalarMatrix flmEqn
142  (
143  fvm::ddt(flm_)
144  + fvm::div(phi(), flm_)
145  ==
146  invT*LM
147  - fvm::Sp(invT, flm_)
148  );
149 
150  flmEqn.relax();
151  flmEqn.solve();
152 
153  bound(flm_, flm0_);
154 
155  volScalarField MM(M && M);
156 
157  fvScalarMatrix fmmEqn
158  (
159  fvm::ddt(fmm_)
160  + fvm::div(phi(), fmm_)
161  ==
162  invT*MM
163  - fvm::Sp(invT, fmm_)
164  );
165 
166  fmmEqn.relax();
167  fmmEqn.solve();
168 
169  bound(fmm_, fmm0_);
170 
171  updateSubGridScaleFields(gradU);
172 }
