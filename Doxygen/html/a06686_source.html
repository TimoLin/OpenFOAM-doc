58 template<class ParcelType>
59 template<class TrackData>
60 void Foam::SprayParcel<ParcelType>::calc
61 (
62  TrackData& td,
63  const scalar dt,
64  const label cellI
65 )
66 {
67  typedef typename TrackData::cloudType::reactingCloudType reactingCloudType;
68  const CompositionModel<reactingCloudType>& composition =
69  td.cloud().composition();
70 
71  // Check if parcel belongs to liquid core
72  if (liquidCore() > 0.5)
73  {
74  // Liquid core parcels should not experience coupled forces
75  td.cloud().forces().setCalcCoupled(false);
76  }
77 
78  // Get old mixture composition
79  const scalarField& Y0(this->Y());
80  scalarField X0(composition.liquids().X(Y0));
81 
82  // Check if we have critical or boiling conditions
83  scalar TMax = composition.liquids().Tc(X0);
84  const scalar T0 = this->T();
85  const scalar pc0 = this->pc_;
86  if (composition.liquids().pv(pc0, T0, X0) >= pc0*0.999)
87  {
88  // Set TMax to boiling temperature
89  TMax = composition.liquids().pvInvert(pc0, X0);
90  }
91 
92  // Set the maximum temperature limit
93  td.cloud().constProps().setTMax(TMax);
94 
95  // Store the parcel properties
96  this->Cp() = composition.liquids().Cp(pc0, T0, X0);
97  sigma_ = composition.liquids().sigma(pc0, T0, X0);
98  const scalar rho0 = composition.liquids().rho(pc0, T0, X0);
99  this->rho() = rho0;
100  const scalar mass0 = this->mass();
101  mu_ = composition.liquids().mu(pc0, T0, X0);
102 
103  ParcelType::calc(td, dt, cellI);
104 
105  if (td.keepParticle)
106  {
107  // Reduce the stripped parcel mass due to evaporation
108  // assuming the number of particles remains unchanged
109  this->ms() -= this->ms()*(mass0 - this->mass())/mass0;
110 
111  // Update Cp, sigma, density and diameter due to change in temperature
112  // and/or composition
113  scalar T1 = this->T();
114  const scalarField& Y1(this->Y());
115  scalarField X1(composition.liquids().X(Y1));
116 
117  this->Cp() = composition.liquids().Cp(this->pc_, T1, X1);
118 
119  sigma_ = composition.liquids().sigma(this->pc_, T1, X1);
120 
121  scalar rho1 = composition.liquids().rho(this->pc_, T1, X1);
122  this->rho() = rho1;
123 
124  mu_ = composition.liquids().mu(this->pc_, T1, X1);
125 
126  scalar d1 = this->d()*cbrt(rho0/rho1);
127  this->d() = d1;
128 
129  if (liquidCore() > 0.5)
130  {
131  calcAtomization(td, dt, cellI);
132 
133  // Preserve the total mass/volume by increasing the number of
134  // particles in parcels due to breakup
135  scalar d2 = this->d();
136  this->nParticle() *= pow3(d1/d2);
137  }
138  else
139  {
140  calcBreakup(td, dt, cellI);
141  }
142  }
143 
144  // Restore coupled forces
145  td.cloud().forces().setCalcCoupled(true);
146 }

149 template<class ParcelType>
150 template<class TrackData>
151 void Foam::SprayParcel<ParcelType>::calcAtomization
152 (
153  TrackData& td,
154  const scalar dt,
155  const label cellI
156 )
157 {
158  typedef typename TrackData::cloudType::reactingCloudType reactingCloudType;
159  const CompositionModel<reactingCloudType>& composition =
160  td.cloud().composition();
161 
162  typedef typename TrackData::cloudType::sprayCloudType sprayCloudType;
163  const AtomizationModel<sprayCloudType>& atomization =
164  td.cloud().atomization();
165 
166  // Average molecular weight of carrier mix - assumes perfect gas
167  scalar Wc = this->rhoc_*specie::RR*this->Tc()/this->pc();
168  scalar R = specie::RR/Wc;
169  scalar Tav = atomization.Taverage(this->T(), this->Tc());
170 
171  // Calculate average gas density based on average temperature
172  scalar rhoAv = this->pc()/(R*Tav);
173 
174  scalar soi = td.cloud().injectors().timeStart();
175  scalar currentTime = td.cloud().db().time().value();
176  const vector& pos = this->position();
177  const vector& injectionPos = this->position0();
178 
179  // Disregard the continous phase when calculating the relative velocity
180  // (in line with the deactivated coupled assumption)
181  scalar Urel = mag(this->U());
182 
183  scalar t0 = max(0.0, currentTime - this->age() - soi);
184  scalar t1 = min(t0 + dt, td.cloud().injectors().timeEnd() - soi);
185 
186  // This should be the vol flow rate from when the parcel was injected
187  scalar volFlowRate = td.cloud().injectors().volumeToInject(t0, t1)/dt;
188 
189  scalar chi = 0.0;
190  if (atomization.calcChi())
191  {
192  chi = this->chi(td, composition.liquids().X(this->Y()));
193  }
194 
195  atomization.update
196  (
197  dt,
198  this->d(),
199  this->liquidCore(),
200  this->tc(),
201  this->rho(),
202  mu_,
203  sigma_,
204  volFlowRate,
205  rhoAv,
206  Urel,
207  pos,
208  injectionPos,
209  td.cloud().pAmbient(),
210  chi,
211  td.cloud().rndGen()
212  );
213 }

216 template<class ParcelType>
217 template<class TrackData>
218 void Foam::SprayParcel<ParcelType>::calcBreakup
219 (
220  TrackData& td,
221  const scalar dt,
222  const label cellI
223 )
224 {
225  typedef typename TrackData::cloudType cloudType;
226  typedef typename cloudType::parcelType parcelType;
227  typedef typename cloudType::forceType forceType;
228 
229  const parcelType& p = static_cast<const parcelType&>(*this);
230  const forceType& forces = td.cloud().forces();
231 
232  if (td.cloud().breakup().solveOscillationEq())
233  {
234  solveTABEq(td, dt);
235  }
236 
237  // Average molecular weight of carrier mix - assumes perfect gas
238  scalar Wc = this->rhoc()*specie::RR*this->Tc()/this->pc();
239  scalar R = specie::RR/Wc;
240  scalar Tav = td.cloud().atomization().Taverage(this->T(), this->Tc());
241 
242  // Calculate average gas density based on average temperature
243  scalar rhoAv = this->pc()/(R*Tav);
244  scalar muAv = this->muc();
245  vector Urel = this->U() - this->Uc();
246  scalar Urmag = mag(Urel);
247  scalar Re = this->Re(this->U(), this->d(), rhoAv, muAv);
248 
249  const scalar mass = p.mass();
250  const forceSuSp Fcp = forces.calcCoupled(p, dt, mass, Re, muAv);
251  const forceSuSp Fncp = forces.calcNonCoupled(p, dt, mass, Re, muAv);
252  this->tMom() = mass/(Fcp.Sp() + Fncp.Sp());
253 
254  const vector g = td.cloud().g().value();
255 
256  scalar parcelMassChild = 0.0;
257  scalar dChild = 0.0;
258  if
259  (
260  td.cloud().breakup().update
261  (
262  dt,
263  g,
264  this->d(),
265  this->tc(),
266  this->ms(),
267  this->nParticle(),
268  this->KHindex(),
269  this->y(),
270  this->yDot(),
271  this->d0(),
272  this->rho(),
273  mu_,
274  sigma_,
275  this->U(),
276  rhoAv,
277  muAv,
278  Urel,
279  Urmag,
280  this->tMom(),
281  dChild,
282  parcelMassChild
283  )
284  )
285  {
286  scalar Re = rhoAv*Urmag*dChild/muAv;
287 
288  // Add child parcel as copy of parent
289  SprayParcel<ParcelType>* child = new SprayParcel<ParcelType>(*this);
290  child->d() = dChild;
291  child->d0() = dChild;
292  const scalar massChild = child->mass();
293  child->mass0() = massChild;
294  child->nParticle() = parcelMassChild/massChild;
295 
296  const forceSuSp Fcp =
297  forces.calcCoupled(*child, dt, massChild, Re, muAv);
298  const forceSuSp Fncp =
299  forces.calcNonCoupled(*child, dt, massChild, Re, muAv);
300 
301  child->age() = 0.0;
302  child->liquidCore() = 0.0;
303  child->KHindex() = 1.0;
304  child->y() = td.cloud().breakup().y0();
305  child->yDot() = td.cloud().breakup().yDot0();
306  child->tc() = 0.0;
307  child->ms() = -GREAT;
308  child->injector() = this->injector();
309  child->tMom() = massChild/(Fcp.Sp() + Fncp.Sp());
310  child->user() = 0.0;
311  child->setCellValues(td, dt, cellI);
312 
313  td.cloud().addParticle(child);
314  }
315 }

318 template<class ParcelType>
319 template<class TrackData>
320 Foam::scalar Foam::SprayParcel<ParcelType>::chi
321 (
322  TrackData& td,
323  const scalarField& X
324 ) const
325 {
326  // Modifications to take account of the flash boiling on primary break-up
327 
328  typedef typename TrackData::cloudType::reactingCloudType reactingCloudType;
329  const CompositionModel<reactingCloudType>& composition =
330  td.cloud().composition();
331 
332  scalar chi = 0.0;
333  scalar T0 = this->T();
334  scalar p0 = this->pc();
335  scalar pAmb = td.cloud().pAmbient();
336 
337  scalar pv = composition.liquids().pv(p0, T0, X);
338 
339  forAll(composition.liquids(), i)
340  {
341  if (pv >= 0.999*pAmb)
342  {
343  // Liquid is boiling - calc boiling temperature
344 
345  const liquidProperties& liq = composition.liquids().properties()[i];
346  scalar TBoil = liq.pvInvert(p0);
347 
348  scalar hl = liq.hl(pAmb, TBoil);
349  scalar iTp = liq.h(pAmb, T0) - pAmb/liq.rho(pAmb, T0);
350  scalar iTb = liq.h(pAmb, TBoil) - pAmb/liq.rho(pAmb, TBoil);
351 
352  chi += X[i]*(iTp - iTb)/hl;
353  }
354  }
355 
356  chi = min(1.0, max(chi, 0.0));
357 
358  return chi;
359 }

362 template<class ParcelType>
363 template<class TrackData>
364 void Foam::SprayParcel<ParcelType>::solveTABEq
365 (
366  TrackData& td,
367  const scalar dt
368 )
369 {
370  const scalar& TABCmu = td.cloud().breakup().TABCmu();
371  const scalar& TABWeCrit = td.cloud().breakup().TABWeCrit();
372  const scalar& TABComega = td.cloud().breakup().TABComega();
373 
374  scalar r = 0.5*this->d();
375  scalar r2 = r*r;
376  scalar r3 = r*r2;
377 
378  // Inverse of characteristic viscous damping time
379  scalar rtd = 0.5*TABCmu*mu_/(this->rho()*r2);
380 
381  // Oscillation frequency (squared)
382  scalar omega2 = TABComega*sigma_/(this->rho()*r3) - rtd*rtd;
383 
384  if (omega2 > 0)
385  {
386  scalar omega = sqrt(omega2);
387  scalar rhoc = this->rhoc();
388  scalar We = this->We(this->U(), r, rhoc, sigma_)/TABWeCrit;
389 
390  scalar y1 = this->y() - We;
391  scalar y2 = this->yDot()/omega;
392 
393  // Update distortion parameters
394  scalar c = cos(omega*dt);
395  scalar s = sin(omega*dt);
396  scalar e = exp(-rtd*dt);
397  y2 = (this->yDot() + y1*rtd)/omega;
398 
399  this->y() = We + e*(y1*c + y2*s);
400  if (this->y() < 0)
401  {
402  this->y() = 0.0;
403  this->yDot() = 0.0;
404  }
405  else
406  {
407  this->yDot() = (We - this->y())*rtd + e*omega*(y2*c - y1*s);
408  }
409  }
410  else
411  {
412  // Reset distortion parameters
413  this->y() = 0;
414  this->yDot() = 0;
415  }
416 }
