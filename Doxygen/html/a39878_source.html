61 (
62  const volScalarField& chi,

64 ) const

67  {

78 (
79  const volScalarField& chi,
80  const volScalarField& fv1
81 ) const
82 {
83  if (ashfordCorrection_)
84  {
85  const volScalarField chiByCv2((1/Cv2_)*chi);

87  return

116 tmp<volScalarField> SpalartAllmaras::fw(const volScalarField& Stilda) const
117 {
118  volScalarField r
119  (
120  min
121  (
122  nuTilda_
123  /(
124  max
125  (
126  Stilda,
127  dimensionedScalar("SMALL", Stilda.dimensions(), SMALL)
128  )
129  *sqr(kappa_*d_)
130  ),
131  scalar(10.0)
132  )
133  );
134  r.boundaryField() == 0.0;
135 
136  const volScalarField g(r + Cw2_*(pow6(r) - r));
137 
138  return g*pow((1.0 + pow6(Cw3_))/(pow6(g) + pow6(Cw3_)), 1.0/6.0);
139 }

421 void SpalartAllmaras::correct()
422 {
423  if (!turbulence_)
424  {
425  // Re-calculate viscosity
426  mut_ = rho_*nuTilda_*fv1(chi());
427  mut_.correctBoundaryConditions();
428 
429  // Re-calculate thermal diffusivity
430  alphat_ = mut_/Prt_;
431  alphat_.correctBoundaryConditions();
432 
433  return;
434  }
435 
436  RASModel::correct();
437 
438  if (mesh_.changing())
439  {
440  d_.correct();
441  }
442 
443  const volScalarField chi(this->chi());
444  const volScalarField fv1(this->fv1(chi));
445 
446  const volScalarField Stilda
447  (
448  fv3(chi, fv1)*::sqrt(2.0)*mag(skew(fvc::grad(U_)))
449  + fv2(chi, fv1)*nuTilda_/sqr(kappa_*d_)
450  );
451 
452  tmp<fvScalarMatrix> nuTildaEqn
453  (
454  fvm::ddt(rho_, nuTilda_)
455  + fvm::div(phi_, nuTilda_)
456  - fvm::laplacian(DnuTildaEff(), nuTilda_)
457  - Cb2_/sigmaNut_*rho_*magSqr(fvc::grad(nuTilda_))
458  ==
459  Cb1_*rho_*Stilda*nuTilda_
460  - fvm::Sp(Cw1_*fw(Stilda)*nuTilda_*rho_/sqr(d_), nuTilda_)
461  );
462 
463  nuTildaEqn().relax();
464  solve(nuTildaEqn);
465  bound(nuTilda_, dimensionedScalar("0", nuTilda_.dimensions(), 0.0));
466  nuTilda_.correctBoundaryConditions();
467 
468  // Re-calculate viscosity
469  mut_.internalField() = fv1*nuTilda_.internalField()*rho_.internalField();
470  mut_.correctBoundaryConditions();
471 
472  // Re-calculate thermal diffusivity
473  alphat_ = mut_/Prt_;
474  alphat_.correctBoundaryConditions();
475 }
