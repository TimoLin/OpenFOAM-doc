278 void RNGkEpsilon::correct()
279 {
280  if (!turbulence_)
281  {
282  // Re-calculate viscosity
283  mut_ = rho_*Cmu_*sqr(k_)/epsilon_;
284  mut_.correctBoundaryConditions();
285 
286  // Re-calculate thermal diffusivity
287  alphat_ = mut_/Prt_;
288  alphat_.correctBoundaryConditions();
289 
290  return;
291  }
292 
293  RASModel::correct();
294 
295  volScalarField divU(fvc::div(phi_/fvc::interpolate(rho_)));
296 
297  if (mesh_.moving())
298  {
299  divU += fvc::div(mesh_.phi());
300  }
301 
302  tmp<volTensorField> tgradU = fvc::grad(U_);
303  volScalarField S2((tgradU() && dev(twoSymm(tgradU()))));
304  tgradU.clear();
305 
306  volScalarField G(GName(), mut_*S2);
307 
308  volScalarField eta(sqrt(mag(S2))*k_/epsilon_);
309  volScalarField eta3(eta*sqr(eta));
310 
311  volScalarField R
312  (
313  ((eta*(-eta/eta0_ + scalar(1)))/(beta_*eta3 + scalar(1)))
314  );
315 
316  // Update epsilon and G at the wall
317  epsilon_.boundaryField().updateCoeffs();
318 
319  // Dissipation equation
320  tmp<fvScalarMatrix> epsEqn
321  (
322  fvm::ddt(rho_, epsilon_)
323  + fvm::div(phi_, epsilon_)
324  - fvm::laplacian(DepsilonEff(), epsilon_)
325  ==
326  (C1_ - R)*G*epsilon_/k_
327  - fvm::SuSp(((2.0/3.0)*C1_ + C3_)*rho_*divU, epsilon_)
328  - fvm::Sp(C2_*rho_*epsilon_/k_, epsilon_)
329  );
330 
331  epsEqn().relax();
332 
333  epsEqn().boundaryManipulate(epsilon_.boundaryField());
334 
335  solve(epsEqn);
336  bound(epsilon_, epsilonMin_);
337 
338 
339  // Turbulent kinetic energy equation
340 
341  tmp<fvScalarMatrix> kEqn
342  (
343  fvm::ddt(rho_, k_)
344  + fvm::div(phi_, k_)
345  - fvm::laplacian(DkEff(), k_)
346  ==
347  G - fvm::SuSp(2.0/3.0*rho_*divU, k_)
348  - fvm::Sp(rho_*epsilon_/k_, k_)
349  );
350 
351  kEqn().relax();
352  solve(kEqn);
353  bound(k_, kMin_);
354 
355 
356  // Re-calculate viscosity
357  mut_ = rho_*Cmu_*sqr(k_)/epsilon_;
358  mut_.correctBoundaryConditions();
359 
360  // Re-calculate thermal diffusivity
361  alphat_ = mut_/Prt_;
362  alphat_.correctBoundaryConditions();
363 }
