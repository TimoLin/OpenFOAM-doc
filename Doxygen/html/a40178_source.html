234 tmp<fvVectorMatrix> LRR::divDevReff(volVectorField& U) const
235 {
236  if (couplingFactor_.value() > 0.0)
237  {
238  return
239  (
240  fvc::div(R_ + couplingFactor_*nut_*fvc::grad(U), "div(R)")
241  + fvc::laplacian
242  (
243  (1.0 - couplingFactor_)*nut_,
244  U,
245  "laplacian(nuEff,U)"
246  )
247  - fvm::laplacian(nuEff(), U)

332 void LRR::correct()
333 {
334  RASModel::correct();
335 
336  if (!turbulence_)
337  {
338  return;
339  }
340 
341  volSymmTensorField P(-twoSymm(R_ & fvc::grad(U_)));
342  volScalarField G(GName(), 0.5*mag(tr(P)));
343 
344  // Update epsilon and G at the wall
345  epsilon_.boundaryField().updateCoeffs();
346 
347  // Dissipation equation
348  tmp<fvScalarMatrix> epsEqn
349  (
350  fvm::ddt(epsilon_)
351  + fvm::div(phi_, epsilon_)
352  //- fvm::laplacian(Ceps*(K/epsilon_)*R, epsilon_)
353  - fvm::laplacian(DepsilonEff(), epsilon_)
354  ==
355  C1_*G*epsilon_/k_
356  - fvm::Sp(C2_*epsilon_/k_, epsilon_)
357  );
358 
359  epsEqn().relax();
360 
361  epsEqn().boundaryManipulate(epsilon_.boundaryField());
362 
363  solve(epsEqn);
364  bound(epsilon_, epsilonMin_);
365 
366 
367  // Reynolds stress equation
368 
369  const fvPatchList& patches = mesh_.boundary();
370 
371  forAll(patches, patchi)
372  {
373  const fvPatch& curPatch = patches[patchi];
374 
375  if (isA<wallFvPatch>(curPatch))
376  {
377  forAll(curPatch, facei)
378  {
379  label faceCelli = curPatch.faceCells()[facei];
380  P[faceCelli] *= min
381  (
382  G[faceCelli]/(0.5*mag(tr(P[faceCelli])) + SMALL),
383  1.0
384  );
385  }
386  }
387  }
388 
389 
390  tmp<fvSymmTensorMatrix> REqn
391  (
392  fvm::ddt(R_)
393  + fvm::div(phi_, R_)
394  //- fvm::laplacian(Cs*(k_/epsilon_)*R_, R_)
395  - fvm::laplacian(DREff(), R_)
396  + fvm::Sp(Clrr1_*epsilon_/k_, R_)
397  ==
398  P
399  - (2.0/3.0*(1 - Clrr1_)*I)*epsilon_
400  - Clrr2_*dev(P)
401  );
402 
403  REqn().relax();
404  solve(REqn);
405 
406  R_.max
407  (
408  dimensionedSymmTensor
409  (
410  "zero",
411  R_.dimensions(),
412  symmTensor
413  (
414  kMin_.value(), -GREAT, -GREAT,
415  kMin_.value(), -GREAT,
416  kMin_.value()
417  )
418  )
419  );
420 
421  k_ = 0.5*tr(R_);
422  bound(k_, kMin_);
423 
424 
425  // Re-calculate viscosity
426  nut_ = Cmu_*sqr(k_)/epsilon_;
427  nut_.correctBoundaryConditions();
428 
429 
430  // Correct wall shear stresses
431 
432  forAll(patches, patchi)
433  {
434  const fvPatch& curPatch = patches[patchi];
435 
436  if (isA<wallFvPatch>(curPatch))
437  {
438  symmTensorField& Rw = R_.boundaryField()[patchi];
439 
440  const scalarField& nutw = nut_.boundaryField()[patchi];
441 
442  const vectorField snGradU(U_.boundaryField()[patchi].snGrad());
443 
444  const vectorField& faceAreas
445  = mesh_.Sf().boundaryField()[patchi];
446 
447  const scalarField& magFaceAreas
448  = mesh_.magSf().boundaryField()[patchi];
449 
450  forAll(curPatch, facei)
451  {
452  // Calculate near-wall velocity gradient
453  tensor gradUw
454  = (faceAreas[facei]/magFaceAreas[facei])*snGradU[facei];
455 
456  // Calculate near-wall shear-stress tensor
457  tensor tauw = -nutw[facei]*2*symm(gradUw);
458 
459  // Reset the shear components of the stress tensor
460  Rw[facei].xy() = tauw.xy();
461  Rw[facei].xz() = tauw.xz();
462  Rw[facei].yz() = tauw.yz();
463  }
464  }
465  }
466 }
