41 // Fill changedFaces (with face labels) and changedFacesInfo (with delta)
42 // This is the initial set of faces from which to start the waves.
43 // Since there might be lots of places with delta jumps we can follow various
44 // strategies for this initial 'seed'.
45 // - start from single cell/face and let FaceCellWave pick up all others
46 // from there. might be quite a few waves before everything settles.
47 // - start from all faces. Lots of initial transfers.
48 // We do something inbetween:
49 // - start from all faces where there is a jump. Since we cannot easily
50 // determine this across coupled patches (cyclic, processor) introduce
51 // all faces of these and let FaceCellWave sort it out.
52 void Foam::smoothDelta::setChangedFaces
53 (
54  const polyMesh& mesh,
55  const volScalarField& delta,
56  DynamicList<label>& changedFaces,
57  DynamicList<deltaData>& changedFacesInfo
58 )
59 {
60  for (label faceI = 0; faceI < mesh.nInternalFaces(); faceI++)
61  {
62  scalar ownDelta = delta[mesh.faceOwner()[faceI]];
63 
64  scalar neiDelta = delta[mesh.faceNeighbour()[faceI]];
65 
66  // Check if owner delta much larger than neighbour delta or vice versa
67 
68  if (ownDelta > maxDeltaRatio_ * neiDelta)
69  {
70  changedFaces.append(faceI);
71  changedFacesInfo.append(deltaData(ownDelta));
72  }
73  else if (neiDelta > maxDeltaRatio_ * ownDelta)
74  {
75  changedFaces.append(faceI);
76  changedFacesInfo.append(deltaData(neiDelta));
77  }
78  }
79 
80  // Insert all faces of coupled patches no matter what. Let FaceCellWave
81  // sort it out.
82  forAll(mesh.boundaryMesh(), patchI)
83  {
84  const polyPatch& patch = mesh.boundaryMesh()[patchI];
85 
86  if (patch.coupled())
87  {
88  forAll(patch, patchFaceI)
89  {
90  label meshFaceI = patch.start() + patchFaceI;
91 
92  scalar ownDelta = delta[mesh.faceOwner()[meshFaceI]];
93 
94  changedFaces.append(meshFaceI);
95  changedFacesInfo.append(deltaData(ownDelta));
96  }
97  }
98  }
99 
100  changedFaces.shrink();
101  changedFacesInfo.shrink();
102 }

105 void Foam::smoothDelta::calcDelta()
106 {
107  const volScalarField& geometricDelta = geometricDelta_();
108 
109  // Fill changed faces with info
110  DynamicList<label> changedFaces(mesh_.nFaces()/100 + 100);
111  DynamicList<deltaData> changedFacesInfo(changedFaces.size());
112 
113  setChangedFaces(mesh_, geometricDelta, changedFaces, changedFacesInfo);
114 
115  // Set initial field on cells.
116  List<deltaData> cellDeltaData(mesh_.nCells());
117 
118  forAll(geometricDelta, cellI)
119  {
120  cellDeltaData[cellI] = geometricDelta[cellI];
121  }
122 
123  // Set initial field on faces.
124  List<deltaData> faceDeltaData(mesh_.nFaces());
125 
126 
127  // Propagate information over whole domain.
128  FaceCellWave<deltaData, scalar> deltaCalc
129  (
130  mesh_,
131  changedFaces,
132  changedFacesInfo,
133  faceDeltaData,
134  cellDeltaData,
135  mesh_.globalData().nTotalCells()+1, // max iterations
136  maxDeltaRatio_
137  );
138 
139  forAll(delta_, cellI)
140  {
141  delta_[cellI] = cellDeltaData[cellI].delta();
142  }
143 }
