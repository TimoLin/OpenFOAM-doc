58 template<class RhoType, class SpType, class SuType>
59 void Foam::MULES::implicitSolve
60 (
61  const RhoType& rho,
62  volScalarField& psi,
63  const surfaceScalarField& phi,
64  surfaceScalarField& phiPsi,
65  const SpType& Sp,
66  const SuType& Su,
67  const scalar psiMax,
68  const scalar psiMin
69 )
70 {
71  const fvMesh& mesh = psi.mesh();
72 
73  const dictionary& MULEScontrols = mesh.solverDict(psi.name());
74 
75  label maxIter
76  (
77  readLabel(MULEScontrols.lookup("maxIter"))
78  );
79 
80  label nLimiterIter
81  (
82  readLabel(MULEScontrols.lookup("nLimiterIter"))
83  );
84 
85  scalar maxUnboundedness
86  (
87  readScalar(MULEScontrols.lookup("maxUnboundedness"))
88  );
89 
90  scalar CoCoeff
91  (
92  readScalar(MULEScontrols.lookup("CoCoeff"))
93  );
94 
95  scalarField allCoLambda(mesh.nFaces());
96 
97  {
98  slicedSurfaceScalarField CoLambda
99  (
100  IOobject
101  (
102  "CoLambda",
103  mesh.time().timeName(),
104  mesh,
105  IOobject::NO_READ,
106  IOobject::NO_WRITE,
107  false
108  ),
109  mesh,
110  dimless,
111  allCoLambda,
112  false // Use slices for the couples
113  );
114 
115  if (phi.dimensions() == dimDensity*dimVelocity*dimArea)
116  {
117  tmp<surfaceScalarField> Cof =
118  mesh.time().deltaT()*mesh.surfaceInterpolation::deltaCoeffs()
119  *mag(phi/interpolate(rho))/mesh.magSf();
120 
121  CoLambda == 1.0/max(CoCoeff*Cof, scalar(1));
122  }
123  else
124  {
125  tmp<surfaceScalarField> Cof =
126  mesh.time().deltaT()*mesh.surfaceInterpolation::deltaCoeffs()
127  *mag(phi)/mesh.magSf();
128 
129  CoLambda == 1.0/max(CoCoeff*Cof, scalar(1));
130  }
131  }
132 
133  scalarField allLambda(allCoLambda);
134  //scalarField allLambda(mesh.nFaces(), 1.0);
135 
136  slicedSurfaceScalarField lambda
137  (
138  IOobject
139  (
140  "lambda",
141  mesh.time().timeName(),
142  mesh,
143  IOobject::NO_READ,
144  IOobject::NO_WRITE,
145  false
146  ),
147  mesh,
148  dimless,
149  allLambda,
150  false // Use slices for the couples
151  );
152 
153  linear<scalar> CDs(mesh);
154  upwind<scalar> UDs(mesh, phi);
155  //fv::uncorrectedSnGrad<scalar> snGrads(mesh);
156 
157  fvScalarMatrix psiConvectionDiffusion
158  (
159  fvm::ddt(rho, psi)
160  + fv::gaussConvectionScheme<scalar>(mesh, phi, UDs).fvmDiv(phi, psi)
161  //- fv::gaussLaplacianScheme<scalar, scalar>(mesh, CDs, snGrads)
162  //.fvmLaplacian(Dpsif, psi)
163  - fvm::Sp(Sp, psi)
164  - Su
165  );
166 
167  surfaceScalarField phiBD(psiConvectionDiffusion.flux());
168 
169  surfaceScalarField& phiCorr = phiPsi;
170  phiCorr -= phiBD;
171 
172  for (label i=0; i<maxIter; i++)
173  {
174  if (i != 0 && i < 4)
175  {
176  allLambda = allCoLambda;
177  }
178 
179  limiter
180  (
181  allLambda,
182  1.0/mesh.time().deltaTValue(),
183  rho,
184  psi,
185  phiBD,
186  phiCorr,
187  Sp,
188  Su,
189  psiMax,
190  psiMin,
191  nLimiterIter
192  );
193 
194  solve
195  (
196  psiConvectionDiffusion + fvc::div(lambda*phiCorr),
197  MULEScontrols
198  );
199 
200  scalar maxPsiM1 = gMax(psi.internalField()) - 1.0;
201  scalar minPsi = gMin(psi.internalField());
202 
203  scalar unboundedness = max(max(maxPsiM1, 0.0), -min(minPsi, 0.0));
204 
205  if (unboundedness < maxUnboundedness)
206  {
207  break;
208  }
209  else
210  {
211  Info<< "MULES: max(" << psi.name() << " - 1) = " << maxPsiM1
212  << " min(" << psi.name() << ") = " << minPsi << endl;
213 
214  phiBD = psiConvectionDiffusion.flux();
215 
216  /*
217  word gammaScheme("div(phi,gamma)");
218  word gammarScheme("div(phirb,gamma)");
219 
220  const surfaceScalarField& phir =
221  mesh.lookupObject<surfaceScalarField>("phir");
222 
223  phiCorr =
224  fvc::flux
225  (
226  phi,
227  psi,
228  gammaScheme
229  )
230  + fvc::flux
231  (
232  -fvc::flux(-phir, scalar(1) - psi, gammarScheme),
233  psi,
234  gammarScheme
235  )
236  - phiBD;
237  */
238  }
239  }
240 
241  phiPsi = psiConvectionDiffusion.flux() + lambda*phiCorr;
242 }
