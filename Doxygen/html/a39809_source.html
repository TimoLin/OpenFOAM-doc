58 dimensionedScalar homogeneousDynOneEqEddy::ck_
59 (
60  const volSymmTensorField& D
61 ) const
62 {
63  volScalarField KK(0.5*(filter_(magSqr(U())) - magSqr(filter_(U()))));
64 
65  volSymmTensorField LL(dev(filter_(sqr(U())) - (sqr(filter_(U())))));
66 
67  volSymmTensorField MM
68  (
69  delta()*(filter_(sqrt(k_)*D) - 2*sqrt(KK + filter_(k_))*filter_(D))
70  );
71 
72  return average(LL && MM)/average(magSqr(MM));
73 }

76 dimensionedScalar homogeneousDynOneEqEddy::ce_
77 (
78  const volSymmTensorField& D
79 ) const
80 {
81  volScalarField KK(0.5*(filter_(magSqr(U())) - magSqr(filter_(U()))));
82 
83  volScalarField mm
84  (
85  pow(KK + filter_(k_), 1.5)/(2*delta()) - filter_(pow(k_, 1.5))/delta()
86  );
87 
88  volScalarField ee
89  (
90  2*delta()*ck_(D)
91  *(
92  filter_(sqrt(k_)*magSqr(D))
93  - 2*sqrt(KK + filter_(k_))*magSqr(filter_(D))
94  )
95  );
96 
97  return average(ee*mm)/average(mm*mm);
98 }

140 void homogeneousDynOneEqEddy::correct(const tmp<volTensorField>& tgradU)
141 {
142  const volTensorField& gradU = tgradU();
143 
144  GenEddyVisc::correct(gradU);
145 
146  volSymmTensorField D(dev(symm(gradU)));
147  volScalarField divU(fvc::div(phi()/fvc::interpolate(rho())));
148  volScalarField G(GName(), 2*muSgs_*(gradU && D));
149 
150  tmp<fvScalarMatrix> kEqn
151  (
152  fvm::ddt(rho(), k_)
153  + fvm::div(phi(), k_)
154  - fvm::laplacian(DkEff(), k_)
155  ==
156  G
157  - fvm::SuSp(2.0/3.0*rho()*divU, k_)
158  - fvm::Sp(ce_(D)*rho()*sqrt(k_)/delta(), k_)
159  );
160 
161  kEqn().relax();
162  kEqn().solve();
163 
164  bound(k_, kMin_);
165 
166  updateSubGridScaleFields(D);
167 }
