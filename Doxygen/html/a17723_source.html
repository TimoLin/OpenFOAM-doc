156 template<class BasicTurbulenceModel>
157 void NicenoKEqn<BasicTurbulenceModel>::correctNut()
158 {
159  const PhaseCompressibleTurbulenceModel<transportModel>& gasTurbulence =
160  this->gasTurbulence();
161 
162  this->nut_ =
163  this->Ck_*sqrt(this->k_)*this->delta()
164  + Cmub_*gasTurbulence.transport().d()*gasTurbulence.alpha()
165  *(mag(this->U_ - gasTurbulence.U()));
166 
167  this->nut_.correctBoundaryConditions();
168 }

171 template<class BasicTurbulenceModel>
172 tmp<volScalarField> NicenoKEqn<BasicTurbulenceModel>::bubbleG() const
173 {
174  const PhaseCompressibleTurbulenceModel<transportModel>& gasTurbulence =
175  this->gasTurbulence();
176 
177  const transportModel& liquid = this->transport();
178  const twoPhaseSystem& fluid = liquid.fluid();
179  const transportModel& gas = fluid.otherPhase(liquid);
180 
181  volScalarField magUr(mag(this->U_ - gasTurbulence.U()));
182 
183  tmp<volScalarField> bubbleG
184  (
185  Cp_*sqr(magUr)*fluid.drag(gas).K()/liquid.rho()
186  );
187 
188  return bubbleG;
189 }

192 template<class BasicTurbulenceModel>
193 tmp<volScalarField>
194 NicenoKEqn<BasicTurbulenceModel>::phaseTransferCoeff() const
195 {
196  const volVectorField& U = this->U_;
197  const alphaField& alpha = this->alpha_;
198  const rhoField& rho = this->rho_;
199 
200  const turbulenceModel& gasTurbulence = this->gasTurbulence();
201 
202  return
203  (
204  max(alphaInversion_ - alpha, scalar(0))
205  *rho
206  *min
207  (
208  this->Ce_*sqrt(gasTurbulence.k())/this->delta(),
209  1.0/U.time().deltaT()
210  )
211  );
212 }

215 template<class BasicTurbulenceModel>
216 tmp<fvScalarMatrix> NicenoKEqn<BasicTurbulenceModel>::kSource() const
217 {
218  const alphaField& alpha = this->alpha_;
219  const rhoField& rho = this->rho_;
220 
221  const PhaseCompressibleTurbulenceModel<transportModel>& gasTurbulence =
222  this->gasTurbulence();
223 
224  const volScalarField phaseTransferCoeff(this->phaseTransferCoeff());
225 
226  return
227  alpha*rho*bubbleG()
228  + phaseTransferCoeff*gasTurbulence.k()
229  - fvm::Sp(phaseTransferCoeff, this->k_);
