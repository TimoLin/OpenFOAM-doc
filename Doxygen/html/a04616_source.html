66 inline bool Foam::fv::option::inTimeLimits(const scalar time) const
67 {
68  return
69  (
70  (timeStart_ < 0)
71  ||
72  (
73  (mesh_.time().value() >= timeStart_)
74  && (mesh_.time().value() <= (timeStart_ + duration_))
75  )
76  );
