119 void Foam::cloudInfo::write()
120 {
121  if (active_)
122  {
123  functionObjectFile::write();
124 
125  label i = 0;
126  forAllConstIter(wordHashSet, names(), iter)
127  {
128  const word& cloudName = iter.key();
129 
130  const kinematicCloud& cloud =
131  obr_.lookupObject<kinematicCloud>(cloudName);
132 
133  label nParcels = returnReduce(cloud.nParcels(), sumOp<label>());
134  scalar massInSystem =
135  returnReduce(cloud.massInSystem(), sumOp<scalar>());
136 
137  if (Pstream::master())
138  {
139  file(i)
140  << obr_.time().value() << token::TAB
141  << nParcels << token::TAB
142  << massInSystem << endl;
143  }
144 
145  i++;
146  }
147  }
148 }
