32  const label localSize,

40  labelList localSizes(Pstream::nProcs(comm), 0);
41  localSizes[Pstream::myProcNo(comm)] = localSize;
42  if (parallel)

44  Pstream::gatherList(localSizes, tag, comm);

48  label offset = 0;
