67 void Foam::patchDist::correct()
68 {
69  // Calculate distance starting from patch faces
70  patchWave wave(mesh(), patchIDs_, correctWalls_);
71 
72  // Transfer cell values from wave into *this
73  transfer(wave.distance());
74 
75  // Transfer values on patches into boundaryField of *this
76  forAll(boundaryField(), patchI)
77  {
78  if (!isA<emptyFvPatchScalarField>(boundaryField()[patchI]))
79  {
80  scalarField& waveFld = wave.patchDistance()[patchI];
81 
82  boundaryField()[patchI].transfer(waveFld);
83  }
84  }
85 
86  // Transfer number of unset values
87  nUnset_ = wave.nUnset();
88 }
