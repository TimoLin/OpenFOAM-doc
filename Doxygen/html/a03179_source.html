47 template<>
48 Foam::tmp<Foam::volVectorField>
49 Foam::fv::faceMDLimitedGrad<Foam::scalar>::calcGrad
50 (
51  const volScalarField& vsf,
52  const word& name
53 ) const
54 {
55  const fvMesh& mesh = vsf.mesh();
56 
57  tmp<volVectorField> tGrad = basicGradScheme_().calcGrad(vsf, name);
58 
59  if (k_ < SMALL)
60  {
61  return tGrad;
62  }
63 
64  volVectorField& g = tGrad();
65 
66  const labelUList& owner = mesh.owner();
67  const labelUList& neighbour = mesh.neighbour();
68 
69  const volVectorField& C = mesh.C();
70  const surfaceVectorField& Cf = mesh.Cf();
71 
72  scalar rk = (1.0/k_ - 1.0);
73 
74  forAll(owner, facei)
75  {
76  label own = owner[facei];
77  label nei = neighbour[facei];
78 
79  scalar vsfOwn = vsf[own];
80  scalar vsfNei = vsf[nei];
81 
82  scalar maxFace = max(vsfOwn, vsfNei);
83  scalar minFace = min(vsfOwn, vsfNei);
84 
85  if (k_ < 1.0)
86  {
87  scalar maxMinFace = rk*(maxFace - minFace);
88  maxFace += maxMinFace;
89  minFace -= maxMinFace;
90  }
91 
92  // owner side
93  cellMDLimitedGrad<scalar>::limitFace
94  (
95  g[own],
96  maxFace - vsfOwn,
97  minFace - vsfOwn,
98  Cf[facei] - C[own]
99  );
100 
101  // neighbour side
102  cellMDLimitedGrad<scalar>::limitFace
103  (
104  g[nei],
105  maxFace - vsfNei,
106  minFace - vsfNei,
107  Cf[facei] - C[nei]
108  );
109  }
110 
111  const volScalarField::GeometricBoundaryField& bsf = vsf.boundaryField();
112 
113  forAll(bsf, patchi)
114  {
115  const fvPatchScalarField& psf = bsf[patchi];
116 
117  const labelUList& pOwner = mesh.boundary()[patchi].faceCells();
118  const vectorField& pCf = Cf.boundaryField()[patchi];
119 
120  if (psf.coupled())
121  {
122  const scalarField psfNei(psf.patchNeighbourField());
123 
124  forAll(pOwner, pFacei)
125  {
126  label own = pOwner[pFacei];
127 
128  scalar vsfOwn = vsf[own];
129  scalar vsfNei = psfNei[pFacei];
130 
131  scalar maxFace = max(vsfOwn, vsfNei);
132  scalar minFace = min(vsfOwn, vsfNei);
133 
134  if (k_ < 1.0)
135  {
136  scalar maxMinFace = rk*(maxFace - minFace);
137  maxFace += maxMinFace;
138  minFace -= maxMinFace;
139  }
140 
141  cellMDLimitedGrad<scalar>::limitFace
142  (
143  g[own],
144  maxFace - vsfOwn,
145  minFace - vsfOwn,
146  pCf[pFacei] - C[own]
147  );
148  }
149  }
150  else if (psf.fixesValue())
151  {
152  forAll(pOwner, pFacei)
153  {
154  label own = pOwner[pFacei];
155 
156  scalar vsfOwn = vsf[own];
157  scalar vsfNei = psf[pFacei];
158 
159  scalar maxFace = max(vsfOwn, vsfNei);
160  scalar minFace = min(vsfOwn, vsfNei);
161 
162  if (k_ < 1.0)
163  {
164  scalar maxMinFace = rk*(maxFace - minFace);
165  maxFace += maxMinFace;
166  minFace -= maxMinFace;
167  }
168 
169  cellMDLimitedGrad<scalar>::limitFace
170  (
171  g[own],
172  maxFace - vsfOwn,
173  minFace - vsfOwn,
174  pCf[pFacei] - C[own]
175  );
176  }
177  }
178  }
179 
180  g.correctBoundaryConditions();
181  gaussGrad<scalar>::correctBoundaryConditions(vsf, g);
182 
183  return tGrad;
184 }
