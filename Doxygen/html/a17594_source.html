414 void LienCubicKELowRe::correct()
415 {
416  RASModel::correct();
417 
418  if (!turbulence_)
419  {
420  return;
421  }
422 
423  if (mesh_.changing())
424  {
425  y_.correct();
426  }
427 
428  tmp<volTensorField> tgradU = fvc::grad(U_);
429  const volTensorField& gradU = tgradU();
430 
431  // generation term
432  tmp<volScalarField> S2 = symm(gradU) && gradU;
433 
434  yStar_ = sqrt(k_)*y_/nu() + SMALL;
435  tmp<volScalarField> Rt = sqr(k_)/(nu()*epsilon_);
436 
437  const volScalarField fMu
438  (
439  (scalar(1) - exp(-Am_*yStar_))
440  /(scalar(1) - exp(-Aepsilon_*yStar_) + SMALL)
441  );
442  const volScalarField f2
443  (
444  scalar(1) - 0.3*exp(-sqr(Rt))
445  );
446 
447  volScalarField G
448  (
449  GName(),
450  Cmu_*fMu*sqr(k_)/epsilon_*S2 - (nonlinearStress_ && gradU)
451  );
452 
453  // Dissipation equation
454  tmp<fvScalarMatrix> epsEqn
455  (
456  fvm::ddt(epsilon_)
457  + fvm::div(phi_, epsilon_)
458  - fvm::laplacian(DepsilonEff(), epsilon_)
459  ==
460  C1_*G*epsilon_/k_
461  // E-term
462  + C2_*f2*pow(Cmu_, 0.75)*pow(k_, scalar(0.5))
463  /(kappa_*y_*(scalar(1) - exp(-Aepsilon_*yStar_)))
464  *exp(-Amu_*sqr(yStar_))*epsilon_
465  - fvm::Sp(C2_*f2*epsilon_/k_, epsilon_)
466  );
467 
468  epsEqn().relax();
469 
470  #include "LienCubicKELowReSetWallDissipation.H"
471  #include "wallDissipationI.H"
472 
473  solve(epsEqn);
474  bound(epsilon_, epsilonMin_);
475 
476 
477  // Turbulent kinetic energy equation
478 
479  tmp<fvScalarMatrix> kEqn
480  (
481  fvm::ddt(k_)
482  + fvm::div(phi_, k_)
483  - fvm::laplacian(DkEff(), k_)
484  ==
485  G
486  - fvm::Sp(epsilon_/k_, k_)
487  );
488 
489  kEqn().relax();
490  solve(kEqn);
491  bound(k_, kMin_);
492 
493 
494  // Re-calculate viscosity
495 
496  eta_ = k_/epsilon_*sqrt(2.0*magSqr(0.5*(gradU + gradU.T())));
497  ksi_ = k_/epsilon_*sqrt(2.0*magSqr(0.5*(gradU - gradU.T())));
498  Cmu_ = 2.0/(3.0*(A1_ + eta_ + alphaKsi_*ksi_));
499  fEta_ = A2_ + pow(eta_, 3.0);
500 
501  C5viscosity_ =
502  - 2.0*pow(Cmu_, 3.0)*pow(k_, 4.0)/pow(epsilon_, 3.0)
503  *(magSqr(gradU + gradU.T()) - magSqr(gradU - gradU.T()));
504 
505  nut_ =
506  Cmu_*fMu*sqr(k_)/epsilon_
507  // C5 term, implicit
508  + max
509  (
510  C5viscosity_,
511  dimensionedScalar("0", C5viscosity_.dimensions(), 0.0)
512  );
513 
514  nonlinearStress_ = symm
515  (
516  // quadratic terms
517  pow(k_, 3.0)/sqr(epsilon_)
518  *(
519  Ctau1_/fEta_
520  *(
521  (gradU & gradU)
522  + (gradU & gradU)().T()
523  )
524  + Ctau2_/fEta_*(gradU & gradU.T())
525  + Ctau3_/fEta_*(gradU.T() & gradU)
526  )
527  // cubic term C4
528  - 20.0*pow(k_, 4.0)/pow(epsilon_, 3.0)
529  *pow(Cmu_, 3.0)
530  *(
531  ((gradU & gradU) & gradU.T())
532  + ((gradU & gradU.T()) & gradU.T())
533  - ((gradU.T() & gradU) & gradU)
534  - ((gradU.T() & gradU.T()) & gradU)
535  )
536  // cubic term C5, explicit part
537  + min
538  (
539  C5viscosity_,
540  dimensionedScalar("0", C5viscosity_.dimensions(), 0.0)
541  )*gradU
542  );
543 }
