75 void Foam::fvMotionSolverEngineMesh::move()
76 {
77  scalar deltaZ = engineDB_.pistonDisplacement().value();
78  Info<< "deltaZ = " << deltaZ << endl;
79 
80  // Position of the top of the static mesh layers above the piston
81  scalar pistonPlusLayers = pistonPosition_.value() + pistonLayers_.value();
82 
83  scalar pistonSpeed = deltaZ/engineDB_.deltaTValue();
84 
85  motionSolver_.pointMotionU().boundaryField()[pistonIndex_] == pistonSpeed;
86 
87  {
88  scalarField linerPoints
89  (
90  boundary()[linerIndex_].patch().localPoints().component(vector::Z)
91  );
92 
93  motionSolver_.pointMotionU().boundaryField()[linerIndex_] ==
94  pistonSpeed*pos(deckHeight_.value() - linerPoints)
95  *(deckHeight_.value() - linerPoints)
96  /(deckHeight_.value() - pistonPlusLayers);
97  }
98 
99  motionSolver_.solve();
100 
101  if (engineDB_.foundObject<surfaceScalarField>("phi"))
102  {
103  surfaceScalarField& phi =
104  const_cast<surfaceScalarField&>
105  (engineDB_.lookupObject<surfaceScalarField>("phi"));
106 
107  const volScalarField& rho =
108  engineDB_.lookupObject<volScalarField>("rho");
109 
110  const volVectorField& U =
111  engineDB_.lookupObject<volVectorField>("U");
112 
113  bool absolutePhi = false;
114  if (moving())
115  {
116  phi += fvc::interpolate(rho)*fvc::meshPhi(rho, U);
117  absolutePhi = true;
118  }
119 
120  movePoints(motionSolver_.curPoints());
121 
122  if (absolutePhi)
123  {
124  phi -= fvc::interpolate(rho)*fvc::meshPhi(rho, U);
125  }
126  }
127  else
128  {
129  movePoints(motionSolver_.curPoints());
130  }
131 
132 
133  pistonPosition_.value() += deltaZ;
134 
135  Info<< "clearance: " << deckHeight_.value() - pistonPosition_.value() << nl
136  << "Piston speed = " << pistonSpeed << " m/s" << endl;
137 }
