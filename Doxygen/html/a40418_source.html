41 tmp<scalarField> nutUTabulatedWallFunctionFvPatchScalarField::calcNut() const
42 {
43  const label patchi = patch().index();
44 
45  const turbulenceModel& turbModel =
46  db().lookupObject<turbulenceModel>("turbulenceModel");
47  const scalarField& y = turbModel.y()[patchi];
48  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
49  const scalarField magUp(mag(Uw.patchInternalField() - Uw));
50  const scalarField magGradU(mag(Uw.snGrad()));
51  const tmp<volScalarField> tnu = turbModel.nu();
52  const volScalarField& nu = tnu();
53  const scalarField& nuw = nu.boundaryField()[patchi];
54 
55  return
56  max
57  (
58  scalar(0),
59  sqr(magUp/(calcUPlus(magUp*y/nuw) + ROOTVSMALL))
60  /(magGradU + ROOTVSMALL)
61  - nuw
62  );
63 }

178 tmp<scalarField> nutUTabulatedWallFunctionFvPatchScalarField::yPlus() const
179 {
180  const label patchi = patch().index();
181 
182  const turbulenceModel& turbModel =
183  db().lookupObject<turbulenceModel>("turbulenceModel");
184  const scalarField& y = turbModel.y()[patchi];
185  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
186  const scalarField magUp(mag(Uw.patchInternalField() - Uw));
187  const tmp<volScalarField> tnu = turbModel.nu();
188  const volScalarField& nu = tnu();
189  const scalarField& nuw = nu.boundaryField()[patchi];
190  const scalarField Rey(magUp*y/nuw);
191 
192  return Rey/(calcUPlus(Rey) + ROOTVSMALL);
193 }
