38 void Foam::solutionControl::read(const bool absTolOnly)
39 {
40  const dictionary& solnDict = this->dict();
41 
42  // Read solution controls
43  nNonOrthCorr_ =
44  solnDict.lookupOrDefault<label>("nNonOrthogonalCorrectors", 0);
45  momentumPredictor_ = solnDict.lookupOrDefault("momentumPredictor", true);
46  transonic_ = solnDict.lookupOrDefault("transonic", false);
47 
48  // Read residual information
49  const dictionary residualDict(solnDict.subOrEmptyDict("residualControl"));
50 
51  DynamicList<fieldData> data(residualControl_);
52 
53  forAllConstIter(dictionary, residualDict, iter)
54  {
55  const word& fName = iter().keyword();
56  const label fieldI = applyToField(fName, false);
57  if (fieldI == -1)
58  {
59  fieldData fd;
60  fd.name = fName.c_str();
61 
62  if (absTolOnly)
63  {
64  fd.absTol = readScalar(residualDict.lookup(fName));
65  fd.relTol = -1;
66  fd.initialResidual = -1;
67  }
68  else
69  {
70  if (iter().isDict())
71  {
72  const dictionary& fieldDict(iter().dict());
73  fd.absTol = readScalar(fieldDict.lookup("tolerance"));
74  fd.relTol = readScalar(fieldDict.lookup("relTol"));
75  fd.initialResidual = 0.0;
76  }
77  else
78  {
79  FatalErrorIn("bool Foam::solutionControl::read()")
80  << "Residual data for " << iter().keyword()
81  << " must be specified as a dictionary"
82  << exit(FatalError);
83  }
84  }
85 
86  data.append(fd);
87  }
88  else
89  {
90  fieldData& fd = data[fieldI];
91  if (absTolOnly)
92  {
93  fd.absTol = readScalar(residualDict.lookup(fName));
94  }
95  else
96  {
97  if (iter().isDict())
98  {
99  const dictionary& fieldDict(iter().dict());
100  fd.absTol = readScalar(fieldDict.lookup("tolerance"));
101  fd.relTol = readScalar(fieldDict.lookup("relTol"));
102  }
103  else
104  {
105  FatalErrorIn("bool Foam::solutionControl::read()")
106  << "Residual data for " << iter().keyword()
107  << " must be specified as a dictionary"
108  << exit(FatalError);
109  }
110  }
111  }
112  }
113 
114  residualControl_.transfer(data);
115 
116  if (debug)
117  {
118  forAll(residualControl_, i)
119  {
120  const fieldData& fd = residualControl_[i];
121  Info<< "residualControl[" << i << "]:" << nl
122  << " name : " << fd.name << nl
123  << " absTol : " << fd.absTol << nl
124  << " relTol : " << fd.relTol << nl
125  << " iniResid : " << fd.initialResidual << endl;
126  }
127  }
128 }

131 Foam::label Foam::solutionControl::applyToField
132 (
133  const word& fieldName,
134  const bool useRegEx
135 ) const
136 {
137  forAll(residualControl_, i)
138  {
139  if (useRegEx && residualControl_[i].name.match(fieldName))
140  {
141  return i;
142  }
143  else if (residualControl_[i].name == fieldName)
144  {
145  return i;
146  }
147  }
148 
149  return -1;
150 }
