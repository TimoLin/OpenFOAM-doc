103 tmp<scalarField> nutLowReWallFunctionFvPatchScalarField::yPlus() const
104 {
105  const label patchi = patch().index();
106  const turbulenceModel& turbModel =
107  db().lookupObject<turbulenceModel>("turbulenceModel");
108  const scalarField& y = turbModel.y()[patchi];
109  const tmp<volScalarField> tnu = turbModel.nu();
110  const volScalarField& nu = tnu();
111  const scalarField& nuw = nu.boundaryField()[patchi];
112  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
113 
114  return y*sqrt(nuw*mag(Uw.snGrad()))/nuw;
115 }
