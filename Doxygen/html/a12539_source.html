165 inline Scalar sqrtSumSqr(const Scalar a, const Scalar b)
166 {
167  Scalar maga = mag(a);
168  Scalar magb = mag(b);
169 
170  if (maga > magb)
171  {
172  return maga*sqrt(1.0 + sqr(magb/maga));
173  }
174  else
175  {
176  return magb < ScalarVSMALL ? 0.0 : magb*sqrt(1.0 + sqr(maga/magb));
177  }
178 }
