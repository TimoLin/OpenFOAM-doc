67 void writeWeights(const polyMesh& mesh)
68 {
69  const polyBoundaryMesh& pbm = mesh.boundaryMesh();
70 
71  const word tmName(mesh.time().timeName());
72 
73  forAll(pbm, patchI)
74  {
75  if (isA<cyclicAMIPolyPatch>(pbm[patchI]))
76  {
77  const cyclicAMIPolyPatch& cpp =
78  refCast<const cyclicAMIPolyPatch>(pbm[patchI]);
79 
80  if (cpp.owner())
81  {
82  Info<< "Calculating AMI weights between owner patch: "
83  << cpp.name() << " and neighbour patch: "
84  << cpp.neighbPatch().name() << endl;
85 
86  const AMIPatchToPatchInterpolation& ami =
87  cpp.AMI();
88 
89  writeWeights
90  (
91  ami.tgtWeightsSum(),
92  cpp.neighbPatch(),
93  "postProcessing",
94  "tgt",
95  tmName
96  );
97  writeWeights
98  (
99  ami.srcWeightsSum(),
100  cpp,
101  "postProcessing",
102  "src",
103  tmName
104  );
105  }
106  }
107  }
108 }

130  }
131 
132  while (runTime.loop())

136  mesh.update();
