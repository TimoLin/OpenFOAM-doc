157 List<Vb::Point> rayShooting::initialPoints() const
158 {
159  // Loop over surface faces
160  const searchableSurfaces& surfaces = geometryToConformTo().geometry();
161  const labelList& surfacesToConformTo = geometryToConformTo().surfaces();
162 
163  const scalar maxRayLength(surfaces.bounds().mag());
164 
165  // Initialise points list
166  label initialPointsSize = 0;
167  forAll(surfaces, surfI)
168  {
169  initialPointsSize += surfaces[surfI].size();
170  }
171 
172  DynamicList<Vb::Point> initialPoints(initialPointsSize);
173 
174  forAll(surfacesToConformTo, surfI)
175  {
176  const searchableSurface& s = surfaces[surfacesToConformTo[surfI]];
177 
178  tmp<pointField> faceCentresTmp(s.coordinates());
179  const pointField& faceCentres = faceCentresTmp();
180 
181  Info<< " Shoot rays from " << s.name() << nl
182  << " nRays = " << faceCentres.size() << endl;
183 
184  forAll(faceCentres, fcI)
185  {
186  const Foam::point& fC = faceCentres[fcI];
187 
188  if
189  (
190  Pstream::parRun()
191  && !decomposition().positionOnThisProcessor(fC)
192  )
193  {
194  continue;
195  }
196 
197  const scalar pert
198  (
199  randomPerturbationCoeff_
200  *cellShapeControls().cellSize(fC)
201  );
202 
203  pointIndexHit surfHitStart;
204  label hitSurfaceStart;
205 
206  // Face centres should be on the surface so search distance can be
207  // small
208  geometryToConformTo().findSurfaceNearest
209  (
210  fC,
211  sqr(pert),
212  surfHitStart,
213  hitSurfaceStart
214  );
215 
216  vectorField normStart(1, vector::min);
217  geometryToConformTo().getNormal
218  (
219  hitSurfaceStart,
220  List<pointIndexHit>(1, surfHitStart),
221  normStart
222  );
223 
224  pointIndexHit surfHitEnd;
225  label hitSurfaceEnd;
226 
227  geometryToConformTo().findSurfaceNearestIntersection
228  (
229  fC - normStart[0]*pert,
230  fC - normStart[0]*maxRayLength,
231  surfHitEnd,
232  hitSurfaceEnd
233  );
234 
235  if (surfHitEnd.hit())
236  {
237  vectorField normEnd(1, vector::min);
238  geometryToConformTo().getNormal
239  (
240  hitSurfaceEnd,
241  List<pointIndexHit>(1, surfHitEnd),
242  normEnd
243  );
244 
245  if ((normStart[0] & normEnd[0]) < 0)
246  {
247  line<point, point> l(fC, surfHitEnd.hitPoint());
248 
249  if (Pstream::parRun())
250  {
251  // Clip the line in parallel
252  pointIndexHit procIntersection =
253  decomposition().findLine
254  (
255  l.start(),
256  l.end()
257  );
258 
259  if (procIntersection.hit())
260  {
261  l =
262  line<point, point>
263  (
264  l.start(),
265  procIntersection.hitPoint()
266  );
267  }
268  }
269 
270  splitLine
271  (
272  l,
273  pert,
274  initialPoints
275  );
276  }
277  }
278  }
279  }
280 
281  return initialPoints.shrink();
282 }
