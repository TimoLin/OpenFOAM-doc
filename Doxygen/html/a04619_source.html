43 void Foam::fv::option::writeData(Ostream& os) const
44 {
45  os.writeKeyword("active") << active_ << token::END_STATEMENT << nl;
46  os.writeKeyword("timeStart") << timeStart_ << token::END_STATEMENT << nl;
47  os.writeKeyword("duration") << duration_ << token::END_STATEMENT << nl;
48  os.writeKeyword("selectionMode")
49  << selectionModeTypeNames_[selectionMode_] << nl;
50 
51  switch (selectionMode_)
52  {
53  case smPoints:
54  {
55  os.writeKeyword("points") << points_
56  << token::END_STATEMENT << nl;
57  break;
58  }
59  case smCellSet:
60  {
61  os.writeKeyword("cellSet") << cellSetName_
62  << token::END_STATEMENT << nl;
63  break;
64  }
65  case smCellZone:
66  {
67  os.writeKeyword("cellZone") << cellSetName_
68  << token::END_STATEMENT << nl;
69  break;
70  }
71  case smAll:
72  {
73  break;
74  }
75  case smMapRegion:
76  {
77  break;
78  }
79  default:
80  {
81  FatalErrorIn("option::writeData(Ostream&) const")
82  << "Unknown selectionMode "
83  << selectionMode_
84  << abort(FatalError);
85  }
86  }
87 }
