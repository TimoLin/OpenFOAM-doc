184 void turbulentHeatFluxTemperatureFvPatchScalarField::updateCoeffs()
185 {
186  if (updated())
187  {
188  return;
189  }
190 
191  const scalarField& alphaEffp =
192  patch().lookupPatchField<volScalarField, scalar>(alphaEffName_);
193 
194  // retrieve (constant) specific heat capacity from transport dictionary
195  const IOdictionary& transportProperties =
196  db().lookupObject<IOdictionary>("transportProperties");
197  const scalar rhoCp0(readScalar(transportProperties.lookup("rhoCp0")));
198 
199  switch (heatSource_)
200  {
201  case hsPower:
202  {
203  const scalar Ap = gSum(patch().magSf());
204  gradient() = q_/(Ap*rhoCp0*alphaEffp);
205  break;
206  }
207  case hsFlux:
208  {
209  gradient() = q_/(rhoCp0*alphaEffp);
210  break;
211  }
212  default:
213  {
214  FatalErrorIn
215  (
216  "turbulentHeatFluxTemperatureFvPatchScalarField"
217  "("
218  "const fvPatch&, "
219  "const DimensionedField<scalar, volMesh>&, "
220  "const dictionary&"
221  ")"
222  ) << "Unknown heat source type. Valid types are: "
223  << heatSourceTypeNames_ << nl << exit(FatalError);
224  }
225  }
226 
227  fixedGradientFvPatchScalarField::updateCoeffs();
228 }
