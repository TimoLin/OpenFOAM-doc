110 void Foam::fv::fixedTemperatureConstraint::setValue
111 (
112  fvMatrix<scalar>& eqn,
113  const label
114 )
115 {
116  const basicThermo& thermo =
117  mesh_.lookupObject<basicThermo>("thermophysicalProperties");
118 
119  if (eqn.psi().name() == thermo.he().name())
120  {
121  switch (mode_)
122  {
123  case tmUniform:
124  {
125  const scalar t = mesh_.time().value();
126  scalarField Tuni(cells_.size(), Tuniform_->value(t));
127  eqn.setValues(cells_, thermo.he(thermo.p(), Tuni, cells_));
128 
129  break;
130  }
131  case tmLookup:
132  {
133  const volScalarField& T =
134  mesh().lookupObject<volScalarField>(TName_);
135 
136  scalarField Tlkp(T, cells_);
137  eqn.setValues(cells_, thermo.he(thermo.p(), Tlkp, cells_));
138 
139  break;
140  }
141  default:
142  {
143  // error handling done by NamedEnum
144  }
