51 void Foam::fv::MRFSource::initialise()
52 {
53  if (selectionMode_ != smCellZone)
54  {
55  FatalErrorIn("void Foam::MRFSource::initialise()")
56  << "The MRF region must be specified as a cellZone. Current "
57  << "selection mode is " << selectionModeTypeNames_[selectionMode_]
58  << exit(FatalError);
59  }
60 
61  mrfPtr_.reset
62  (
63  new MRFZone
64  (
65  name_,
66  mesh_,
67  coeffs_,
68  cellSetName_
69  )
70  );
71 
72  const volVectorField& U = mesh_.lookupObject<volVectorField>(UName_);
73 
74  mrfPtr_->correctBoundaryVelocity(const_cast<volVectorField&>(U));
75 
76  fieldNames_.setSize(1, UName_);
77  applied_.setSize(1, false);
78 }
