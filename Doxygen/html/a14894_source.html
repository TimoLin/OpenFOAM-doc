85

92  {

94  }
95  else

98  }

168 void sixDoFRigidBodyDisplacementPointPatchVectorField::updateCoeffs()
169 {
170  if (this->updated())
171  {
172  return;
173  }
174 
175  if (lookupGravity_ < 0)
176  {
177  if (db().foundObject<uniformDimensionedVectorField>("g"))
178  {
179  if (lookupGravity_ == -2)
180  {
181  FatalErrorIn
182  (
183  "void sixDoFRigidBodyDisplacementPointPatchVectorField"
184  "::updateCoeffs()"
185  )
186  << "Specifying the value of g in this boundary condition "
187  << "when g is available from the database is considered "
188  << "a fatal error to avoid the possibility of inconsistency"
189  << exit(FatalError);
190  }
191  else
192  {
193  lookupGravity_ = 1;
194  }
195  }
196  else
197  {
198  lookupGravity_ = 0;
199  }
200  }
201 
202  const polyMesh& mesh = this->dimensionedInternalField().mesh()();
203  const Time& t = mesh.time();
204  const pointPatch& ptPatch = this->patch();
205 
206  // Store the motion state at the beginning of the time-step
207  if (curTimeIndex_ != t.timeIndex())
208  {
209  motion_.newTime();
210  curTimeIndex_ = t.timeIndex();
211  }
212 
213  // Patch force data is valid for the current positions, so
214  // calculate the forces on the motion object from this data, then
215  // update the positions
216 
217  motion_.updatePosition(t.deltaTValue(), t.deltaT0Value());
218 
219  dictionary forcesDict;
220 
221  forcesDict.add("type", forces::typeName);
222  forcesDict.add("patches", wordList(1, ptPatch.name()));
223  forcesDict.add("rhoInf", rhoInf_);
224  forcesDict.add("rhoName", rhoName_);
225  forcesDict.add("CofR", motion_.centreOfRotation());
226 
227  forces f("forces", db(), forcesDict);
228 
229  f.calcForcesMoment();
230 
231  // Get the forces on the patch faces at the current positions
232 
233  if (lookupGravity_ == 1)
234  {
235  uniformDimensionedVectorField g =
236  db().lookupObject<uniformDimensionedVectorField>("g");
237 
238  g_ = g.value();
239  }
240 
241  // scalar ramp = min(max((t.value() - 5)/10, 0), 1);
242  scalar ramp = 1.0;
243 
244  motion_.updateAcceleration
245  (
246  ramp*(f.forceEff() + motion_.mass()*g_),
247  ramp*(f.momentEff() + motion_.mass()*(motion_.momentArm() ^ g_)),
248  t.deltaTValue()
249  );
250 
251  Field<vector>::operator=
252  (
253  motion_.transform(initialPoints_) - initialPoints_
254  );
255 
256  fixedValuePointPatchField<vector>::updateCoeffs();
257 }

260 void sixDoFRigidBodyDisplacementPointPatchVectorField::write(Ostream& os) const
261 {
262  pointPatchField<vector>::write(os);
263 
264  os.writeKeyword("rhoName") << rhoName_ << token::END_STATEMENT << nl;
265 
266  if (rhoName_ == "rhoInf")
267  {
268  os.writeKeyword("rhoInf") << rhoInf_ << token::END_STATEMENT << nl;
269  }
270 
271  if (lookupGravity_ == 0 || lookupGravity_ == -2)
272  {
273  os.writeKeyword("g") << g_ << token::END_STATEMENT << nl;
274  }
275 
276  motion_.write(os);
277 
278  initialPoints_.writeEntry("initialPoints", os);
279 
280  writeEntry("value", os);
281 }
