47 tmp<volScalarField> kOmegaSST::F1(const volScalarField& CDkOmega) const
48 {
49  tmp<volScalarField> CDkOmegaPlus = max
50  (
51  CDkOmega,
52  dimensionedScalar("1.0e-10", dimless/sqr(dimTime), 1.0e-10)
53  );
54 
55  tmp<volScalarField> arg1 = min
56  (
57  min
58  (
59  max
60  (
61  (scalar(1)/betaStar_)*sqrt(k_)/(omega_*y_),
62  scalar(500)*nu()/(sqr(y_)*omega_)
63  ),
64  (4*alphaOmega2_)*k_/(CDkOmegaPlus*sqr(y_))
65  ),
66  scalar(10)
67  );
68 
69  return tanh(pow4(arg1));
70 }

398 void kOmegaSST::correct()
399 {
400  RASModel::correct();
401 
402  if (!turbulence_)
403  {
404  return;
405  }
406 
407  if (mesh_.changing())
408  {
409  y_.correct();
410  }
411 
412  const volScalarField S2(2*magSqr(symm(fvc::grad(U_))));
413  volScalarField G(GName(), nut_*S2);
414 
415  // Update omega and G at the wall
416  omega_.boundaryField().updateCoeffs();
417 
418  const volScalarField CDkOmega
419  (
420  (2*alphaOmega2_)*(fvc::grad(k_) & fvc::grad(omega_))/omega_
421  );
422 
423  const volScalarField F1(this->F1(CDkOmega));
424 
425  // Turbulent frequency equation
426  tmp<fvScalarMatrix> omegaEqn
427  (
428  fvm::ddt(omega_)
429  + fvm::div(phi_, omega_)
430  - fvm::laplacian(DomegaEff(F1), omega_)
431  ==
432  gamma(F1)
433  *min(S2, (c1_/a1_)*betaStar_*omega_*max(a1_*omega_, b1_*F23()*sqrt(S2)))
434  - fvm::Sp(beta(F1)*omega_, omega_)
435  - fvm::SuSp
436  (
437  (F1 - scalar(1))*CDkOmega/omega_,
438  omega_
439  )
440  );
441 
442  omegaEqn().relax();
443 
444  omegaEqn().boundaryManipulate(omega_.boundaryField());
445 
446  solve(omegaEqn);
447  bound(omega_, omegaMin_);
448 
449  // Turbulent kinetic energy equation
450  tmp<fvScalarMatrix> kEqn
451  (
452  fvm::ddt(k_)
453  + fvm::div(phi_, k_)
454  - fvm::laplacian(DkEff(F1), k_)
455  ==
456  min(G, c1_*betaStar_*k_*omega_)
457  - fvm::Sp(betaStar_*omega_, k_)
458  );
459 
460  kEqn().relax();
461  solve(kEqn);
462  bound(k_, kMin_);
463 
464 
465  // Re-calculate viscosity
466  nut_ = a1_*k_/max(a1_*omega_, b1_*F23()*sqrt(S2));
467  nut_.correctBoundaryConditions();
468 }
