289 template<class CloudType>
290 Foam::scalarField Foam::CompositionModel<CloudType>::X
291 (
292  const label phaseI,
293  const scalarField& Y
294 ) const
295 {
296  const phaseProperties& props = phaseProps_[phaseI];
297  scalarField X(Y.size(), 0.0);
298  scalar WInv = 0.0;
299  switch (props.phase())
300  {
301  case phaseProperties::GAS:
302  {
303  forAll(Y, i)
304  {
305  label gid = props.globalIds()[i];
306  WInv += Y[i]/thermo_.carrier().W(gid);
307  X[i] = Y[i]/thermo_.carrier().W(gid);
308  }
309  break;
310  }
311  case phaseProperties::LIQUID:
312  {
313  forAll(Y, i)
314  {
315  label gid = props.globalIds()[i];
316  WInv += Y[i]/thermo_.liquids().properties()[gid].W();
317  X[i] += Y[i]/thermo_.liquids().properties()[gid].W();
318  }
319  break;
320  }
321  default:
322  {
323  FatalErrorIn
324  (
325  "Foam::scalarField Foam::CompositionModel<CloudType>::X"
326  "("
327  "const label, "
328  "const scalarField&"
329  ") const"
330  ) << "Only possible to convert gas and liquid mass fractions"
331  << abort(FatalError);
332  }
333  }
334 
335  tmp<scalarField> tfld = X/WInv;
336  return tfld();
337 }

390 Foam::scalar Foam::CompositionModel<CloudType>::H
391 (
392  const label phaseI,
393  const scalarField& Y,
394  const scalar p,
395  const scalar T
396 ) const
397 {
398  const phaseProperties& props = phaseProps_[phaseI];
399  scalar HMixture = 0.0;
400  switch (props.phase())
401  {
402  case phaseProperties::GAS:
403  {
404  forAll(Y, i)
405  {
406  label gid = props.globalIds()[i];
407  HMixture += Y[i]*thermo_.carrier().Ha(gid, p, T);
408  }
409  break;
410  }
411  case phaseProperties::LIQUID:
412  {
413  forAll(Y, i)
414  {
415  label gid = props.globalIds()[i];
416  HMixture += Y[i]*thermo_.liquids().properties()[gid].h(p, T);
417  }
418  break;
419  }
420  case phaseProperties::SOLID:
421  {
422  forAll(Y, i)
423  {
424  label gid = props.globalIds()[i];
425  HMixture +=
426  Y[i]
427  *(
428  thermo_.solids().properties()[gid].Hf()
429  + thermo_.solids().properties()[gid].Cp()*T
430  );
431  }
432  break;
433  }
434  default:
435  {
436  FatalErrorIn
437  (
438  "Foam::scalar Foam::CompositionModel<CloudType>::H"
439  "("
440  " const label, "
441  " const scalarField&, "
442  " const scalar, "
443  " const scalar"
444  ") const"
445  ) << "Unknown phase enumeration" << abort(FatalError);
446  }
447  }
448 
449  return HMixture;
