57 const Foam::labelList& Foam::primitiveMesh::edgeFaces
58 (
59  const label edgeI,
60  DynamicList<label>& storage
61 ) const
62 {
63  if (hasEdgeFaces())
64  {
65  return edgeFaces()[edgeI];
66  }
67  else
68  {
69  // Use the fact that pointEdges are sorted in incrementing edge order
70  const edge& e = edges()[edgeI];
71  const labelList& pFaces0 = pointFaces()[e[0]];
72  const labelList& pFaces1 = pointFaces()[e[1]];
73 
74  label i0 = 0;
75  label i1 = 0;
76 
77  storage.clear();
78 
79  while (i0 < pFaces0.size() && i1 < pFaces1.size())
80  {
81  if (pFaces0[i0] < pFaces1[i1])
82  {
83  ++i0;
84  }
85  else if (pFaces0[i0] > pFaces1[i1])
86  {
87  ++i1;
88  }
89  else
90  {
91  // Equal. Append.
92  storage.append(pFaces0[i0]);
93  ++i0;
94  ++i1;
95  }
96  }
97 
98  return storage;
99  }
100 }
