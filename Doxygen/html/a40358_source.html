41 scalar nutkRoughWallFunctionFvPatchScalarField::fnRough
42 (
43  const scalar KsPlus,
44  const scalar Cs
45 ) const
46 {
47  // Return fn based on non-dimensional roughness height
48 
49  if (KsPlus < 90.0)
50  {
51  return pow
52  (
53  (KsPlus - 2.25)/87.75 + Cs*KsPlus,
54  sin(0.4258*(log(KsPlus) - 0.811))
55  );
56  }
57  else
58  {
59  return (1.0 + Cs*KsPlus);
60  }
61 }

64 tmp<scalarField> nutkRoughWallFunctionFvPatchScalarField::calcNut() const
65 {
66  const label patchi = patch().index();
67 
68  const turbulenceModel& turbModel =
69  db().lookupObject<turbulenceModel>("turbulenceModel");
70  const scalarField& y = turbModel.y()[patchi];
71  const tmp<volScalarField> tk = turbModel.k();
72  const volScalarField& k = tk();
73  const tmp<volScalarField> tnu = turbModel.nu();
74  const volScalarField& nu = tnu();
75  const scalarField& nuw = nu.boundaryField()[patchi];
76 
77  const scalar Cmu25 = pow025(Cmu_);
78 
79  tmp<scalarField> tnutw(new scalarField(*this));
80  scalarField& nutw = tnutw();
81 
82  forAll(nutw, faceI)
83  {
84  label faceCellI = patch().faceCells()[faceI];
85 
86  scalar uStar = Cmu25*sqrt(k[faceCellI]);
87  scalar yPlus = uStar*y[faceI]/nuw[faceI];
88  scalar KsPlus = uStar*Ks_[faceI]/nuw[faceI];
89 
90  scalar Edash = E_;
91  if (KsPlus > 2.25)
92  {
93  Edash /= fnRough(KsPlus, Cs_[faceI]);
94  }
95 
96  scalar limitingNutw = max(nutw[faceI], nuw[faceI]);
97 
98  // To avoid oscillations limit the change in the wall viscosity
99  // which is particularly important if it temporarily becomes zero
100  nutw[faceI] =
101  max
102  (
103  min
104  (
105  nuw[faceI]
106  *(yPlus*kappa_/log(max(Edash*yPlus, 1+1e-4)) - 1),
107  2*limitingNutw
108  ), 0.5*limitingNutw
109  );
110 
111  if (debug)
112  {
113  Info<< "yPlus = " << yPlus
114  << ", KsPlus = " << KsPlus
115  << ", Edash = " << Edash
116  << ", nutw = " << nutw[faceI]
117  << endl;
118  }
119  }
120 
121  return tnutw;
122 }
