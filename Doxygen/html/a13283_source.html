77 template<class Type>
78 Type Foam::fieldValues::cellSource::processValues
79 (
80  const Field<Type>& values,
81  const scalarField& V,
82  const scalarField& weightField
83 ) const
84 {
85  Type result = pTraits<Type>::zero;
86  switch (operation_)
87  {
88  case opSum:
89  {
90  result = sum(values);
91  break;
92  }
93  case opSumMag:
94  {
95  result = sum(cmptMag(values));
96  break;
97  }
98  case opAverage:
99  {
100  result = sum(values)/values.size();
101  break;
102  }
103  case opWeightedAverage:
104  {
105  result = sum(values)/sum(weightField);
106  break;
107  }
108  case opVolAverage:
109  {
110  result = sum(values*V)/sum(V);
111  break;
112  }
113  case opVolIntegrate:
114  {
115  result = sum(values*V);
116  break;
117  }
118  case opMin:
119  {
120  result = min(values);
121  break;
122  }
123  case opMax:
124  {
125  result = max(values);
126  break;
127  }
128  case opCoV:
129  {
130  Type meanValue = sum(values*V)/sum(V);
131 
132  const label nComp = pTraits<Type>::nComponents;
133 
134  for (direction d=0; d<nComp; ++d)
135  {
136  scalarField vals(values.component(d));
137  scalar mean = component(meanValue, d);
138  scalar& res = setComponent(result, d);
139 
140  res = sqrt(sum(V*sqr(vals - mean))/sum(V))/mean;
141  }
142 
143  break;
144  }
145  default:
146  {
147  // Do nothing
148  }
149  }
150 
151  return result;
152 }

157 template<class Type>
158 bool Foam::fieldValues::cellSource::writeValues(const word& fieldName)
159 {
160  const bool ok = validField<Type>(fieldName);
161 
162  if (ok)
163  {
164  Field<Type> values(setFieldValues<Type>(fieldName));
165  scalarField V(filterField(mesh().V()));
166  scalarField weightField(values.size(), 1.0);
167 
168  if (weightFieldName_ != "none")
169  {
170  weightField = setFieldValues<scalar>(weightFieldName_, true);
171  }
172 
173  // Combine onto master
174  combineFields(values);
175  combineFields(V);
176  combineFields(weightField);
177 
178  // apply weight field
179  values *= weightField;
180 
181  if (Pstream::master())
182  {
183  Type result = processValues(values, V, weightField);
184 
185  // add to result dictionary, over-writing any previous entry
186  resultDict_.add(fieldName, result, true);
187 
188  if (valueOutput_)
189  {
190  IOField<Type>
191  (
192  IOobject
193  (
194  fieldName + "_" + sourceTypeNames_[source_] + "-"
195  + sourceName_,
196  obr_.time().timeName(),
197  obr_,
198  IOobject::NO_READ,
199  IOobject::NO_WRITE
200  ),
201  values
202  ).write();
203  }
204 
205 
206  file()<< tab << result;
207 
208  Info(log_)<< " " << operationTypeNames_[operation_]
209  << "(" << sourceName_ << ") of " << fieldName
210  << " = " << result << endl;
211  }
212  }
213 
214  return ok;
215 }
