41 tmp<scalarField> mutUSpaldingWallFunctionFvPatchScalarField::calcUTau
42 (
43  const scalarField& magGradU
44 ) const
45 {
46  const turbulenceModel& turbModel =
47  db().lookupObject<turbulenceModel>("turbulenceModel");
48 
49  const scalarField& y = turbModel.y()[patch().index()];
50 
51  const fvPatchVectorField& Uw =
52  turbModel.U().boundaryField()[patch().index()];
53 
54  scalarField magUp(mag(Uw.patchInternalField() - Uw));
55 
56  const fvPatchScalarField& rhow =
57  turbModel.rho().boundaryField()[patch().index()];
58 
59  const fvPatchScalarField& muw =
60  turbModel.mu().boundaryField()[patch().index()];
61  const scalarField& mutw = *this;
62 
63  tmp<scalarField> tuTau(new scalarField(patch().size(), 0.0));
64  scalarField& uTau = tuTau();
65 
66  forAll(mutw, faceI)
67  {
68  scalar ut =
69  sqrt((mutw[faceI] + muw[faceI])*magGradU[faceI]/rhow[faceI]);
70 
71  if (ut > ROOTVSMALL)
72  {
73  int iter = 0;
74  scalar err = GREAT;
75 
76  do
77  {
78  scalar kUu = min(kappa_*magUp[faceI]/ut, 50);
79  scalar fkUu = exp(kUu) - 1 - kUu*(1 + 0.5*kUu);
80 
81  scalar f =
82  - ut*y[faceI]/(muw[faceI]/rhow[faceI])
83  + magUp[faceI]/ut
84  + 1/E_*(fkUu - 1.0/6.0*kUu*sqr(kUu));
85 
86  scalar df =
87  y[faceI]/(muw[faceI]/rhow[faceI])
88  + magUp[faceI]/sqr(ut)
89  + 1/E_*kUu*fkUu/ut;
90 
91  scalar uTauNew = ut + f/df;
92  err = mag((ut - uTauNew)/ut);
93  ut = uTauNew;
94 
95  } while (ut > ROOTVSMALL && err > 0.01 && ++iter < 10);
96 
97  uTau[faceI] = max(0.0, ut);
98  }
99  }
100 
101  return tuTau;
102 }

105 tmp<scalarField> mutUSpaldingWallFunctionFvPatchScalarField::calcMut() const
106 {
107  const label patchi = patch().index();
108 
109  const turbulenceModel& turbModel =
110  db().lookupObject<turbulenceModel>("turbulenceModel");
111 
112  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
113  const scalarField magGradU(mag(Uw.snGrad()));
114  const scalarField& rhow = turbModel.rho().boundaryField()[patchi];
115  const scalarField& muw = turbModel.mu().boundaryField()[patchi];
116
