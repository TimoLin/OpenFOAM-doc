158 void v2WallFunctionFvPatchScalarField::updateCoeffs()
159 {
160  if (updated())
161  {
162  return;
163  }
164 
165  const label patchi = patch().index();
166 
167  const turbulenceModel& turbulence = db().lookupObject<turbulenceModel>
168  (
169  IOobject::groupName
170  (
171  turbulenceModel::propertiesName,
172  dimensionedInternalField().group()
173  )
174  );
175  const scalarField& y = turbulence.y()[patchi];
176 
177  const tmp<volScalarField> tk = turbulence.k();
178  const volScalarField& k = tk();
179 
180  const tmp<scalarField> tnuw = turbulence.nu(patchi);
181  const scalarField& nuw = tnuw();
182 
183  const scalar Cmu25 = pow025(Cmu_);
184 
185  scalarField& v2 = *this;
186 
187  // Set v2 wall values
188  forAll(v2, faceI)
189  {
190  label faceCellI = patch().faceCells()[faceI];
191 
192  scalar uTau = Cmu25*sqrt(k[faceCellI]);
193 
194  scalar yPlus = uTau*y[faceI]/nuw[faceI];
195 
196  if (yPlus > yPlusLam_)
197  {
198  scalar Cv2 = 0.193;
199  scalar Bv2 = -0.94;
200  v2[faceI] = Cv2/kappa_*log(yPlus) + Bv2;
201  }
202  else
203  {
204  scalar Cv2 = 0.193;
205  v2[faceI] = Cv2*pow4(yPlus);
206  }
207 
208  v2[faceI] *= sqr(uTau);
209  }
210 
211  fixedValueFvPatchField<scalar>::updateCoeffs();
212 
213  // TODO: perform averaging for cells sharing more than one boundary face
214 }
