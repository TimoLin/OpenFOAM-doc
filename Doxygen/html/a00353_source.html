81 {
82  if (undisplacedPoints_.size() != nPoints())
83  {
84  FatalIOErrorIn
85  (
86  "solidBodyMotionFvMesh::solidBodyMotionFvMesh(const IOobject&)",
87  dynamicMeshCoeffs_
88  ) << "Read " << undisplacedPoints_.size()

92  }
93 
94  word cellZoneName =
95  dynamicMeshCoeffs_.lookupOrDefault<word>("cellZone", "none");
96 
97  word cellSetName =
98  dynamicMeshCoeffs_.lookupOrDefault<word>("cellSet", "none");
99 
100  if ((cellZoneName != "none") && (cellSetName != "none"))
101  {
102  FatalIOErrorIn
103  (
104  "solidBodyMotionFvMesh::solidBodyMotionFvMesh(const IOobject&)",
105  dynamicMeshCoeffs_
106  )
107  << "Either cellZone OR cellSet can be supplied, but not both. "
108  << "If neither is supplied, all cells will be included"
109  << exit(FatalIOError);
110  }
111 
112 
113  labelList cellIDs;
114  if (cellZoneName != "none")
115  {
116  Info<< "Applying solid body motion to cellZone " << cellZoneName
117  << endl;
118 
119  label zoneID = cellZones().findZoneID(cellZoneName);
120 
121  if (zoneID == -1)
122  {
123  FatalErrorIn
124  (
125  "solidBodyMotionFvMesh::solidBodyMotionFvMesh(const IOobject&)"
126  )
127  << "Unable to find cellZone " << cellZoneName
128  << ". Valid celLZones are:"
129  << cellZones().names()
130  << exit(FatalError);
131  }
132 
133  cellIDs = cellZones()[zoneID];
134  }
135 
136  if (cellSetName != "none")
137  {
138  Info<< "Applying solid body motion to cellSet " << cellSetName
139  << endl;
140 
141  cellSet set(*this, cellSetName);
142 
143  cellIDs = set.toc();
144  }
145 
146  label nCells = returnReduce(cellIDs.size(), sumOp<label>());
147  moveAllCells_ = nCells == 0;
148 
149  if (moveAllCells_)
150  {
151  Info<< "Applying solid body motion to entire mesh" << endl;
152  }
153  else
154  {
155  // collect point IDs of points in cell zone
156 
157  boolList movePts(nPoints(), false);
158 
159  forAll(cellIDs, i)
160  {
161  label cellI = cellIDs[i];
162  const cell& c = cells()[cellI];
163  forAll(c, j)
164  {
165  const face& f = faces()[c[j]];
166  forAll(f, k)
167  {
168  label pointI = f[k];
169  movePts[pointI] = true;
170  }
171  }
172  }
173 
174  syncTools::syncPointList(*this, movePts, orEqOp<bool>(), false);
175 
176  DynamicList<label> ptIDs(nPoints());
177  forAll(movePts, i)
178  {
179  if (movePts[i])
180  {
181  ptIDs.append(i);
182  }
183  }
184 
185  pointIDs_.transfer(ptIDs);
186  }
187 }

198 bool Foam::solidBodyMotionFvMesh::update()
199 {
200  static bool hasWarned = false;
201 
202  if (moveAllCells_)
203  {
204  fvMesh::movePoints
205  (
206  transform
207  (
208  SBMFPtr_().transformation(),
209  undisplacedPoints_
210  )
211  );
212  }
213  else
214  {
215  pointField transformedPts(undisplacedPoints_);
216 
217  UIndirectList<point>(transformedPts, pointIDs_) =
218  transform
219  (
220  SBMFPtr_().transformation(),
221  pointField(transformedPts, pointIDs_)
222  );
223 
224  fvMesh::movePoints(transformedPts);
225  }
226 
227 
228  if (foundObject<volVectorField>(UName_))
229  {
230  const volVectorField& U = lookupObject<volVectorField>(UName_);
231 
232  const_cast<volVectorField&>(U).correctBoundaryConditions();
233  }
234  else if (!hasWarned)
235  {
236  hasWarned = true;
237 
238  WarningIn("solidBodyMotionFvMesh::update()")
239  << "Did not find volVectorField " << UName_
240  << " Not updating " << UName_ << "boundary conditions."
241  << endl;
242  }
243 
244  return true;
245 }
