138 tmp<fvVectorMatrix> GenSGSStress::divDevReff
139 (
140  volVectorField& U
141 ) const
142 {

144  {
145  return
146  (

148  + fvc::laplacian

151  )
152  - fvm::laplacian(nuEff(), U)

174

183  )

185  );
186  }
187  else
188  {
189  return
190  (
191  fvc::div(rho*B_)
192  + fvc::laplacian(rho*nuSgs_, U, "laplacian(muEff,U)")
193  - fvm::laplacian(muEff, U)
