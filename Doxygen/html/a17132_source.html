104 void convectiveHeatTransferFvPatchScalarField::updateCoeffs()
105 {
106  if (updated())
107  {
108  return;
109  }
110 
111  const label patchi = patch().index();
112 
113  const turbulenceModel& turbModel =
114  db().lookupObject<turbulenceModel>("turbulenceModel");
115  const scalarField alphaEffw(turbModel.alphaEff(patchi));
116  const scalarField& muw = turbModel.mu().boundaryField()[patchi];
117  const scalarField& rhow = turbModel.rho().boundaryField()[patchi];
118  const vectorField& Uc = turbModel.U();
119  const vectorField& Uw = turbModel.U().boundaryField()[patchi];
120  const scalarField& Tw = turbModel.thermo().T().boundaryField()[patchi];
121  const scalarField& pw = turbModel.thermo().p().boundaryField()[patchi];
122  const scalarField Cpw(turbModel.thermo().Cp(pw, Tw, patchi));
123 
124  const scalarField kappaw(Cpw*alphaEffw);
125  const scalarField Pr(muw*Cpw/kappaw);
126 
127  scalarField& htc = *this;
128  forAll(htc, faceI)
129  {
130  label faceCellI = patch().faceCells()[faceI];
131 
132  scalar Re = rhow[faceI]*mag(Uc[faceCellI] - Uw[faceI])*L_/muw[faceI];
133 
134  if (Re < 5.0E+05)
135  {
136  htc[faceI] = 0.664*sqrt(Re)*cbrt(Pr[faceI])*kappaw[faceI]/L_;
137  }
138  else
139  {
140  htc[faceI] = 0.037*pow(Re, 0.8)*cbrt(Pr[faceI])*kappaw[faceI]/L_;
141  }
142  }
143 
144  fixedValueFvPatchScalarField::updateCoeffs();
145 }
