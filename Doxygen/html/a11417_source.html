34 void Foam::pointPatchMapper::calcAddressing() const
35 {
36  if
37  (
38  directAddrPtr_
39  || interpolationAddrPtr_
40  || weightsPtr_
41  )
42  {
43  FatalErrorIn
44  (
45  "void pointPatchMapper::calcAddressing() const"
46  ) << "Addressing already calculated"
47  << abort(FatalError);
48  }
49 
50  hasUnmapped_ = false;
51 
52  if (direct())
53  {
54  // Direct mapping.
55  directAddrPtr_ = new labelList(mpm_.patchPointMap()[patch_.index()]);
56  labelList& addr = *directAddrPtr_;
57 
58  forAll(addr, i)
59  {
60  if (addr[i] < 0)
61  {
62  hasUnmapped_ = true;
63  }
64  }
65  }
66  else
67  {
68  // Interpolative mapping.
69 
70  // NOTE: Incorrect:
71  // FOR NOW only takes first patch point instead of averaging all
72  // patch points. Problem is we don't know what points were in the patch
73  // for points that were merged.
74 
75  interpolationAddrPtr_ = new labelListList(size());
76  labelListList& addr = *interpolationAddrPtr_;
77 
78  weightsPtr_ = new scalarListList(addr.size());
79  scalarListList& w = *weightsPtr_;
80 
81  const labelList& ppm = mpm_.patchPointMap()[patch_.index()];
82 
83  forAll(ppm, i)
84  {
85  if (ppm[i] >= 0)
86  {
87  addr[i] = labelList(1, ppm[i]);
88  w[i] = scalarList(1, 1.0);
89  }
90  else
91  {
92  // Inserted point.
94  //addr[i] = labelList(1, label(0));
95  //w[i] = scalarList(1, 1.0);
96  hasUnmapped_ = true;
97  }
98  }
99  }
100 }
