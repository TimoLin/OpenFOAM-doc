144 void uncoupledSixDoFRigidBodyDisplacementPointPatchVectorField::updateCoeffs()
145 {
146  if (this->updated())
147  {
148  return;
149  }
150 
151  const polyMesh& mesh = this->dimensionedInternalField().mesh()();
152  const Time& t = mesh.time();
153 
154  // Store the motion state at the beginning of the time-step
155  if (curTimeIndex_ != t.timeIndex())
156  {
157  motion_.newTime();
158  curTimeIndex_ = t.timeIndex();
159  }
160 
161  motion_.updatePosition(t.deltaTValue(), t.deltaT0Value());
162 
163  vector gravity = vector::zero;
164 
165  if (db().foundObject<uniformDimensionedVectorField>("g"))
166  {
167  uniformDimensionedVectorField g =
168  db().lookupObject<uniformDimensionedVectorField>("g");
169 
170  gravity = g.value();
171  }
172 
173  // Do not modify the accelerations, except with gravity, where available
174  motion_.updateAcceleration
175  (
176  gravity*motion_.mass(),
177  vector::zero,
178  t.deltaTValue()
179  );
180 
181  Field<vector>::operator=
182  (
183  motion_.transform(initialPoints_) - initialPoints_
184  );
185 
186  fixedValuePointPatchField<vector>::updateCoeffs();
187 }
