50 //- Specialisation of the jump-condition for the pressure
51 template<>
52 void Foam::porousBafflePressureFvPatchField<Foam::scalar>::updateCoeffs()
53 {
54  if (updated())
55  {
56  return;
57  }
58 
59  const label patchI = patch().index();
60 
61  const surfaceScalarField& phi =
62  db().lookupObject<surfaceScalarField>("phi");
63 
64  const fvsPatchField<scalar>& phip =
65  patch().patchField<surfaceScalarField, scalar>(phi);
66 
67  scalarField Un(phip/patch().magSf());
68 
69  if (phi.dimensions() == dimensionSet(0, 3, -1, 0, 0))
70  {
71  const incompressible::turbulenceModel& turbModel =
72  db().lookupObject<incompressible::turbulenceModel>
73  (
74  "turbulenceModel"
75  );
76 
77  const scalarField nu = turbModel.nu()().boundaryField()[patchI];
78 
79  scalarField magUn(mag(Un));
80 
81  jump_ = -sign(Un)*(D_*nu + I_*0.5*magUn)*magUn*length_;
82  }
83  else
84  {
85  const compressible::turbulenceModel& turbModel =
86  db().lookupObject<compressible::turbulenceModel>
87  (
88  "turbulenceModel"
89  );
90 
91  const scalarField mu = turbModel.mu().boundaryField()[patchI];
92 
93  const scalarField rhow =
94  patch().lookupPatchField<volScalarField, scalar>("rho");
95 
96  Un /= rhow;
97  scalarField magUn(mag(Un));
98 
99  jump_ = -sign(Un)*(D_*mu + I_*0.5*rhow*magUn)*magUn*length_;
100  }
101 
102  if (debug)
103  {
104  scalar avePressureJump = gAverage(jump_);
105  scalar aveVelocity = gAverage(mag(Un));
106 
107  Info<< patch().boundaryMesh().mesh().name() << ':'
108  << patch().name() << ':'
109  << " Average pressure drop :" << avePressureJump
110  << " Average velocity :" << aveVelocity
111  << endl;
112  }
113 
114  fixedJumpFvPatchField<scalar>::updateCoeffs();
115 }
