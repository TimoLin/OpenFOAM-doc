40 void Foam::cubeRootVolDelta::calcDelta()
41 {
42  label nD = mesh().nGeometricD();
43 
44  if (nD == 3)
45  {
46  delta_.internalField() = deltaCoeff_*pow(mesh().V(), 1.0/3.0);
47  }
48  else if (nD == 2)
49  {
50  WarningIn("cubeRootVolDelta::calcDelta()")
51  << "Case is 2D, LES is not strictly applicable\n"
52  << endl;
53 
54  const Vector<label>& directions = mesh().geometricD();
55 
56  scalar thickness = 0.0;
57  for (direction dir=0; dir<directions.nComponents; dir++)
58  {
59  if (directions[dir] == -1)
60  {
61  thickness = mesh().bounds().span()[dir];
62  break;
63  }
64  }
65 
66  delta_.internalField() = deltaCoeff_*sqrt(mesh().V()/thickness);
67  }
68  else
69  {
70  FatalErrorIn("cubeRootVolDelta::calcDelta()")
71  << "Case is not 3D or 2D, LES is not applicable"
72  << exit(FatalError);
73  }
74 }
