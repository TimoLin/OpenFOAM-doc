46 Foam::fileName Foam::functionObjectFile::baseFileDir() const
47 {
48  fileName baseDir = obr_.time().path();
49 
50  if (Pstream::parRun())
51  {
52  // Put in undecomposed case (Note: gives problems for
53  // distributed data running)
54  baseDir = baseDir/".."/outputPrefix;
55  }
56  else
57  {
58  baseDir = baseDir/outputPrefix;
59  }
60 
61  // append mesh name if not default region
62  if (isA<polyMesh>(obr_))
63  {
64  const polyMesh& mesh = refCast<const polyMesh>(obr_);
65  if (mesh.name() != polyMesh::defaultRegion)
66  {
67  baseDir = baseDir/mesh.name();
68  }
69  }
70 
71  return baseDir;
72 }

81 void Foam::functionObjectFile::createFiles()
82 {
83  if (Pstream::master())
84  {
85  const word startTimeName =
86  obr_.time().timeName(obr_.time().startTime().value());
87 
88  label i = 0;
89  forAllConstIter(wordHashSet, names_, iter)
90  {
91  if (!filePtrs_.set(i))
92  {
93  fileName outputDir(baseFileDir()/prefix_/startTimeName);
94 
95  mkDir(outputDir);
96 
97  word fName(iter.key());
98 
99  // check if file already exists
100  IFstream is(outputDir/(fName + ".dat"));
101  if (is.good())
102  {
103  fName = fName + "_" + obr_.time().timeName();
104  }
105 
106  filePtrs_.set(i, new OFstream(outputDir/(fName + ".dat")));
107 
108  initStream(filePtrs_[i]);
109 
110  writeFileHeader(i);
111 
112  i++;
113  }
114  }
115  }
116 }

246 Foam::OFstream& Foam::functionObjectFile::file()
247 {
248  if (!Pstream::master())
249  {
250  FatalErrorIn("Foam::OFstream& Foam::functionObjectFile::file()")
251  << "Request for file() can only be done by the master process"
252  << abort(FatalError);
253  }
254 
255  if (filePtrs_.size() != 1)
256  {
257  WarningIn("Foam::Ostream& Foam::functionObjectFile::file()")
258  << "Requested single file, but multiple files are present"
259  << endl;
260  }
261 
262  if (!filePtrs_.set(0))
263  {
264  FatalErrorIn("Foam::OFstream& Foam::functionObjectFile::file()")
265  << "File pointer at index " << 0 << " not allocated"
266  << abort(FatalError);
267  }
268 
269  return filePtrs_[0];
270 }
