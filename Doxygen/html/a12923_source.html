68 void Foam::regExp::set(const char* pattern, const bool ignoreCase) const
69 {
70  clear();
71 
72  // avoid NULL pointer and zero-length patterns
73  if (pattern && *pattern)
74  {
75  preg_ = new regex_t;
76 
77  int cflags = REG_EXTENDED;
78  if (ignoreCase)
79  {
80  cflags |= REG_ICASE;
81  }
82 
83  int err = regcomp(preg_, pattern, cflags);
84 
85  if (err != 0)

97  }

162 bool Foam::regExp::match(const string& str, List<string>& groups) const
163 {
164  if (preg_ && str.size())
165  {
166  size_t nmatch = ngroups() + 1;
167  regmatch_t pmatch[nmatch];
168 
169  // also verify that the entire string was matched
170  // pmatch[0] is the entire match
171  // pmatch[1..] are the (...) sub-groups
172  if
173  (
174  regexec(preg_, str.c_str(), nmatch, pmatch, 0) == 0
175  && (pmatch[0].rm_so == 0 && pmatch[0].rm_eo == label(str.size()))
176  )
177  {
178  groups.setSize(ngroups());
179  label groupI = 0;
180 
181  for (size_t matchI = 1; matchI < nmatch; matchI++)
182  {
183  if (pmatch[matchI].rm_so != -1 && pmatch[matchI].rm_eo != -1)
184  {
185  groups[groupI] = str.substr
186  (
187  pmatch[matchI].rm_so,
188  pmatch[matchI].rm_eo - pmatch[matchI].rm_so
189  );
190  }
191  else
192  {
193  groups[groupI].clear();
194  }
195  groupI++;
196  }
197 
198  return true;
199  }
200  }
201 
202  groups.clear();
203  return false;
204 }
