42 template<class Container, class T>
43 void Pstream::exchange
44 (
45  const List<Container>& sendBufs,
46  List<Container>& recvBufs,
47  labelListList& sizes,
48  const int tag,
49  const label comm,
50  const bool block
51 )
52 {
53  if (!contiguous<T>())
54  {
55  FatalErrorIn
56  (
57  "Pstream::exchange(..)"
58  ) << "Continuous data only." << Foam::abort(FatalError);
59  }
60 
61  if (sendBufs.size() != UPstream::nProcs(comm))
62  {
63  FatalErrorIn
64  (
65  "Pstream::exchange(..)"
66  ) << "Size of list:" << sendBufs.size()
67  << " does not equal the number of processors:"
68  << UPstream::nProcs(comm)
69  << Foam::abort(FatalError);
70  }
71 
72  sizes.setSize(UPstream::nProcs(comm));
73  labelList& nsTransPs = sizes[UPstream::myProcNo(comm)];
74  nsTransPs.setSize(UPstream::nProcs(comm));
75 
76  forAll(sendBufs, procI)
77  {
78  nsTransPs[procI] = sendBufs[procI].size();
79  }
80 
81  // Send sizes across. Note: blocks.
82  combineReduce(sizes, UPstream::listEq(), tag, comm);
83 
84  if (UPstream::nProcs(comm) > 1)
85  {
86  label startOfRequests = Pstream::nRequests();
87 
88  // Set up receives
89  // ~~~~~~~~~~~~~~~
90 
91  recvBufs.setSize(sendBufs.size());
92  forAll(sizes, procI)
93  {
94  label nRecv = sizes[procI][UPstream::myProcNo(comm)];
95 
96  if (procI != Pstream::myProcNo(comm) && nRecv > 0)
97  {
98  recvBufs[procI].setSize(nRecv);
99  UIPstream::read
100  (
101  UPstream::nonBlocking,
102  procI,
103  reinterpret_cast<char*>(recvBufs[procI].begin()),
104  nRecv*sizeof(T),
105  tag,
106  comm
107  );
108  }
109  }
110 
111 
112  // Set up sends
113  // ~~~~~~~~~~~~
114 
115  forAll(sendBufs, procI)
116  {
117  if (procI != Pstream::myProcNo(comm) && sendBufs[procI].size() > 0)
118  {
119  if
120  (
121  !UOPstream::write
122  (
123  UPstream::nonBlocking,
124  procI,
125  reinterpret_cast<const char*>(sendBufs[procI].begin()),
126  sendBufs[procI].size()*sizeof(T),
127  tag,
128  comm
129  )
130  )
131  {
132  FatalErrorIn("Pstream::exchange(..)")
133  << "Cannot send outgoing message. "
134  << "to:" << procI << " nBytes:"
135  << label(sendBufs[procI].size()*sizeof(T))
136  << Foam::abort(FatalError);
137  }
138  }
139  }
140 
141 
142  // Wait for all to finish
143  // ~~~~~~~~~~~~~~~~~~~~~~
144 
145  if (block)
146  {
147  Pstream::waitRequests(startOfRequests);
148  }
149  }
150 
151  // Do myself
152  recvBufs[Pstream::myProcNo(comm)] = sendBufs[Pstream::myProcNo(comm)];
153 }
