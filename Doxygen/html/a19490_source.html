64  IOdictionary solutionDict
65  (
66  IOobject
67  (
68  "fvSolution",
69  runTime.system(),
70  runTime,
71  IOobject::MUST_READ_IF_MODIFIED,
72  IOobject::NO_WRITE,
73  false
74  )
75  );
76 
77  label nChanged = 0;
78  entry* e = solutionDict.lookupEntryPtr("solvers", false, false);
79  if (e && e->isDict())
80  {
81  nChanged = solution::upgradeSolverDict(e->dict(), true);
82  }
83 
84  Info<< nChanged << " solver settings changed" << nl << endl;
85  if (nChanged)
86  {
87  if (args.optionFound("test"))
88  {
89  Info<< "-test option: no changes made" << nl << endl;
90  }
91  else
92  {
93  if (mvBak(solutionDict.objectPath(), "old"))
94  {
95  Info<< "Backup to "
96  << (solutionDict.objectPath() + ".old") << nl;
97  }
98 
99  solutionDict.writeObject
100  (
101  IOstream::ASCII,
102  IOstream::currentVersion,
103  IOstream::UNCOMPRESSED
104  );
105 
106  Info<< "Write to "
107  << solutionDict.objectPath() << nl << endl;
108  }
109  }
110 
111  return 0;
112 }
