103 tmp<scalarField> mutLowReWallFunctionFvPatchScalarField::yPlus() const
104 {
105  const label patchi = patch().index();
106  const turbulenceModel& turbModel =
107  db().lookupObject<turbulenceModel>("turbulenceModel");
108  const scalarField& y = turbModel.y()[patchi];
109  const scalarField nuw
110  (
111  turbModel.mu().boundaryField()[patchi]
112  /turbModel.rho().boundaryField()[patchi]
113  );
114  const fvPatchVectorField& Uw = turbModel.U().boundaryField()[patchi];
115 
116  return y*sqrt(nuw*mag(Uw.snGrad()))/nuw;
117 }
