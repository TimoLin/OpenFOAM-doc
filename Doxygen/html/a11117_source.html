63 void Foam::data::setSolverPerformance
64 (
65  const word& name,
66  const solverPerformance& sp
67 ) const
68 {
69  dictionary& dict = const_cast<dictionary&>(solverPerformanceDict());
70 
71  List<solverPerformance> perfs;
72 
73  if (prevTimeIndex_ != this->time().timeIndex())
74  {
75  // reset solver performance between iterations
76  prevTimeIndex_ = this->time().timeIndex();
77  dict.clear();
78  }
79  else
80  {
81  dict.readIfPresent(name, perfs);
82  }
83 
84  // append to list
85  perfs.setSize(perfs.size()+1, sp);
86 
87  dict.set(name, perfs);
88 }
